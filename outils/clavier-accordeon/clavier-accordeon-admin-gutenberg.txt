/* ─── Note name conversion ───────────────────────────── */
const FR2EN={
  'Do':'C','Ré':'D','Re':'D','Mi':'E','Fa':'F','Sol':'G','La':'A','Si':'B',
  'Do#':'C#','Ré#':'D#','Fa#':'F#','Sol#':'G#','La#':'A#',
  'Réb':'Db','Mib':'Eb','Solb':'Gb','Lab':'Ab','Sib':'Bb','Dob':'Cb',
};
const EN2FR={
  'C':'Do','D':'Ré','E':'Mi','F':'Fa','G':'Sol','A':'La','B':'Si',
  'C#':'Do#','D#':'Ré#','F#':'Fa#','G#':'Sol#','A#':'La#',
  'Db':'Réb','Eb':'Mib','Gb':'Solb','Ab':'Lab','Bb':'Sib','Cb':'Dob',
};
function displayNote(note){
  if(!note) return '';
  if(lang==='fr') return note;
  // Handle slash-separated individual notes (e.g. "La3/Do#4/Mi4")
  if(note.includes('/')){return note.split('/').map(n=>displayNote(n.trim())).join('/');}
  // Try full word match first for chords like "Sol Maj"
  const m=note.match(/^(.+?)(\s+(?:Maj|Min|7|V|dim|aug|m7|maj7))?(\d?)$/);
  if(!m) return note;
  const root=m[1].trim(), qual=(m[2]||'').trim(), oct=m[3]||'';
  const en=FR2EN[root];
  if(en) return en+oct+(qual?' '+qual:'');
  return note;
}
/* Returns array of constituent note-names (FR, no octave) for a chord string.
   e.g. "La Maj" → ['La','Do#','Mi'],  "Fa Min" → ['Fa','Lab','Do']
   Returns null for non-chord strings (bare notes). */
function chordNotes(chord){
  if(!chord)return null;
  // Already in note-list format ("La3/Do#4/Mi4") — return the individual notes
  if(chord.includes('/'))return chord.split('/').map(n=>n.trim());
  const m=chord.match(/^(.+?)\s+(Maj|Min|7|m7|maj7|dim|aug|V)$/);
  if(!m)return null;
  const root=m[1].trim(), quality=m[2];
  const FR_CHROMA={'Do':0,'Do#':1,'Réb':1,'Ré':2,'Re':2,'Ré#':3,'Mib':3,'Mi':4,'Fa':5,
    'Fa#':6,'Solb':6,'Sol':7,'Sol#':8,'Lab':8,'La':9,'La#':10,'Sib':10,'Si':11,'Dob':11};
  const SHARP=['Do','Do#','Ré','Ré#','Mi','Fa','Fa#','Sol','Sol#','La','La#','Si'];
  const FLAT =['Do','Réb','Ré','Mib','Mi','Fa','Solb','Sol','Lab','La','Sib','Si'];
  const IMAP ={'Maj':[0,4,7],'Min':[0,3,7],'7':[0,4,7,10],'m7':[0,3,7,10],
    'maj7':[0,4,7,11],'dim':[0,3,6],'aug':[0,4,8],'V':[0,7]};
  const idx=FR_CHROMA[root]; if(idx===undefined)return null;
  const ivs=IMAP[quality]; if(!ivs)return null;
  // Use flat spelling for "flat" roots and for minor-quality intervals (m3, m6, m7, dim5)
  const rootFlat=((root.length>2?root.endsWith('b'):false))||root==='Réb';
  return ivs.map(i=>{
    const preferFlat=rootFlat||[3,6,8,10].includes(i);
    return (preferFlat?FLAT:SHARP)[(idx+i)%12];
  });
}
function storageNote(note){
  // Convert EN→FR for storage (picker uses current lang)
  if(lang==='fr') return note;
  const m=note.match(/^([A-G][#b]?)(\d?)(\s+.*)?$/);
  if(!m) return note;
  const fr=EN2FR[m[1]];
  if(fr) return fr+(m[2]||'')+(m[3]||'');
  return note;
}

const SEMI_FR={Do:0,Ré:2,Re:2,Mi:4,Fa:5,Sol:7,La:9,Si:11};
const SEMI_EN={C:0,D:2,E:4,F:5,G:7,A:9,B:11};
function noteToMidi(s){
  if(!s) return null;
  let m=s.trim().match(/^(Do|Ré|Re|Mi|Fa|Sol|La|Si)([#b]?)(\d+)$/);
  if(m){
    const base=SEMI_FR[m[1]]; if(base===undefined) return null;
    return (parseInt(m[3])+1)*12+base+(m[2]==='#'?1:m[2]==='b'?-1:0);
  }
  m=s.trim().match(/^([A-G])([#b]?)(\d+)$/);
  if(m){
    const base=SEMI_EN[m[1]]; if(base===undefined) return null;
    return (parseInt(m[3])+1)*12+base+(m[2]==='#'?1:m[2]==='b'?-1:0);
  }
  return null;
}
const MIDI_TO_FR=['Do','Réb','Ré','Mib','Mi','Fa','Solb','Sol','Lab','La','Sib','Si'];
function midiToFr(midi){if(midi===null||midi===undefined)return'';const oct=Math.floor(midi/12)-1;return MIDI_TO_FR[((midi%12)+12)%12]+oct;}
function shiftNote(n,semi){const m=noteToMidi(n);return m!==null?midiToFr(m+semi):'';}

function renderPresetOptions(){
  const sel=document.getElementById('preset-select');
  if(!sel)return;
  // Group by category
  const cats={};
  _presetIndex.forEach(p=>{(cats[p.category]||(cats[p.category]=[])).push(p);});
  const customOpt=sel.querySelector('option[value="custom"]');
  // Remove previously inserted options
  sel.querySelectorAll('option[data-preset]').forEach(o=>o.remove());
  Object.entries(cats).forEach(([cat,items])=>{
    const og=document.createElement('optgroup');og.label=cat;
    items.forEach(p=>{
      const o=document.createElement('option');
      o.value=p.id;o.textContent=p.label;o.dataset.preset='1';
      og.appendChild(o);
    });
    sel.insertBefore(og,customOpt);
  });
}

async function fetchPresetData(id){
  if(_presetCache[id])return _presetCache[id];
  const base = PRESETS_BASE_URL || document.querySelector('base')?.href || location.href;
  const url = PRESETS_BASE_URL
    ? new URL(`${id}.json`, PRESETS_BASE_URL.replace(/\/?$/, '/')).href
    : new URL(`presets/${id}.json`, base).href;
  const r=await fetch(url);
  if(!r.ok)throw new Error(`Preset not found: ${id}`);
  const data=await r.json();
  _presetCache[id]=data;
  return data;
}

  CFG.lhRows=[{nb:Math.max(1,CFG.notes.basses.length),offset:0}];
  document.getElementById('nb-rangees').value=p.nbRangees;
  syncGaucheInputs();
  renderRangeesCfg();renderLhRowsCfg();generateInputGrid();onAnyChange();
}

let _rowDragSrc=null;
function initRowDragDrop(){
  const wrap=document.getElementById('rangees-cfg');
  if(!wrap)return;
  wrap.querySelectorAll('.rangee-cfg').forEach((el,idx)=>{
    el.dataset.rowIdx=idx;
    el.addEventListener('dragstart',e=>{
      _rowDragSrc=el;
      e.dataTransfer.effectAllowed='move';
      setTimeout(()=>el.classList.add('row-dragging'),0);
    });
    el.addEventListener('dragend',()=>{
      el.classList.remove('row-dragging');
      wrap.querySelectorAll('.rangee-cfg').forEach(c=>c.classList.remove('row-drag-over'));
      _rowDragSrc=null;
    });
    el.addEventListener('dragover',e=>{
      e.preventDefault();e.dataTransfer.dropEffect='move';
      if(_rowDragSrc&&_rowDragSrc!==el){
        wrap.querySelectorAll('.rangee-cfg').forEach(c=>c.classList.remove('row-drag-over'));
        el.classList.add('row-drag-over');
      }
    });
    el.addEventListener('dragleave',()=>el.classList.remove('row-drag-over'));
    el.addEventListener('drop',e=>{
      e.preventDefault();
      if(!_rowDragSrc||_rowDragSrc===el)return;
      const si=parseInt(_rowDragSrc.dataset.rowIdx),ti=parseInt(el.dataset.rowIdx);
      if(isNaN(si)||isNaN(ti)||si===ti)return;

      const nd=CFG.notes.droite.splice(si,1)[0];CFG.notes.droite.splice(ti,0,nd||[]);
      renderRangeesCfg();generateInputGrid();onAnyChange();
    });
  });
}

/* ── Drag-and-swap columns in bass/chord grid ── */
let _dragSrc=null;
function initBtnCellDragDrop(){
  document.querySelectorAll('#note-grid .btn-grid').forEach(grid=>{
    grid.querySelectorAll('.btn-cell').forEach(cell=>{
      cell.draggable=true;
      cell.addEventListener('dragstart',e=>{
        _dragSrc=cell;
        e.dataTransfer.effectAllowed='move';
        setTimeout(()=>cell.classList.add('dragging'),0);
      });
      cell.addEventListener('dragend',()=>{
        cell.classList.remove('dragging');
        grid.querySelectorAll('.btn-cell').forEach(c=>c.classList.remove('drag-over'));
        _dragSrc=null;
      });
      cell.addEventListener('dragover',e=>{
        e.preventDefault();e.dataTransfer.dropEffect='move';
        if(_dragSrc?(_dragSrc!==cell?_dragSrc.closest('.btn-grid')===grid:false):false){
          grid.querySelectorAll('.btn-cell').forEach(c=>c.classList.remove('drag-over'));
          cell.classList.add('drag-over');
        }
      });
      cell.addEventListener('dragleave',()=>cell.classList.remove('drag-over'));
      cell.addEventListener('drop',e=>{
        e.preventDefault();
        if(!_dragSrc||_dragSrc===cell)return;
        const srcGrid=_dragSrc.closest('.btn-grid');
        if(srcGrid!==grid)return;
        // Determine indices
        const cells=[...grid.querySelectorAll('.btn-cell')];
        const si=cells.indexOf(_dragSrc),ti=cells.indexOf(cell);
        if(si<0||ti<0||si===ti)return;
        // Get the note key from the row-wrap id (e.g. "rw-g-basses" → "basses")
        const rowWrap=grid.closest('.row-wrap');
        const rowId=rowWrap?rowWrap.id:'';
        const keyMatch=rowId.match(/^rw-g-(.+)$/);
        if(!keyMatch)return;
        const key=keyMatch[1];
        if(!CFG.notes[key])return;
        // Swap in CFG
        const arr=CFG.notes[key];
        if(si<arr.length?ti<arr.length:false){
          const tmp=arr[si];arr[si]=arr[ti];arr[ti]=tmp;
        }
        generateInputGrid();
        onAnyChange();
      });
    });
  });
}

  const v=input.value.trim();
  if(v){
    const m=v.match(/^(.+?)([2-6])$/);
    if(m){pickerNote=m[1];pickerOctave=m[2];}else pickerNote=v;
  }
  showPickerTab('notes',document.querySelector('.picker-tab-btn:first-child'));
  updatePickerHighlights();updatePickerPreview();
  const pk=document.getElementById('note-picker');pk.style.display='block';
  const rect=input.getBoundingClientRect(),sy=window.scrollY,sx=window.scrollX;
  let top=sy+rect.bottom+6,left=sx+rect.left;
  const pw=pk.offsetWidth||310,ph=pk.offsetHeight||300;
  if(left+pw>sx+window.innerWidth-10)left=sx+window.innerWidth-pw-10;
  if(left<8)left=8;
  if(top+ph>sy+window.innerHeight-10)top=sy+rect.top-ph-6;
  pk.style.top=top+'px';pk.style.left=left+'px';
  input.classList.add('picker-open');
}
function hidePicker(){
  document.getElementById('note-picker').style.display='none';
  if(pickerInput){pickerInput.classList.remove('picker-open');pickerInput=null;}
}
function showPickerTab(name,btn){
  document.querySelectorAll('.picker-tab-btn').forEach(b=>b.classList.remove('active'));
  if(btn)btn.classList.add('active');
  document.getElementById('picker-tab-notes').classList.toggle('active',name==='notes');
}
function pickerSetNote(n){pickerNote=n;updatePickerHighlights();updatePickerPreview();if(pickerNote?pickerOctave:false)pickerInsert();}
function pickerSetOctave(o){pickerOctave=o;updatePickerHighlights();updatePickerPreview();if(pickerNote?pickerOctave:false)pickerInsert();}
function pickerInsertDirect(text){
  if(!pickerInput)return;
  pickerInput.value=storageNote(text);
  pickerInput.dispatchEvent(new Event('input',{bubbles:true}));
  hidePicker();onAnyChange();
}
function pickerInsert(){
  if(!pickerInput)return;
  const v=pickerNote+pickerOctave;if(!v)return;
  pickerInput.value=storageNote(v);
  pickerInput.dispatchEvent(new Event('input',{bubbles:true}));
  hidePicker();onAnyChange();
}
function pickerClear(){
  if(!pickerInput)return;
  pickerInput.value='';pickerInput.dispatchEvent(new Event('input',{bubbles:true}));
  pickerNote='';pickerOctave='';updatePickerHighlights();updatePickerPreview();hidePicker();onAnyChange();
}
function updatePickerHighlights(){
  document.querySelectorAll('#picker-tab-notes .pn').forEach(b=>b.classList.toggle('selected',b.textContent===pickerNote));
  document.querySelectorAll('.po').forEach(b=>b.classList.toggle('selected',b.textContent===pickerOctave));
}
function updatePickerPreview(){
  const txt=pickerNote+pickerOctave;
  document.getElementById('picker-preview-text').textContent=txt||t('selectNoteOctave');
  const showMidi=document.getElementById('show-midi')?.checked;
  const badge=document.getElementById('picker-midi-badge');
  if(badge){
    const m=txt?noteToMidi(storageNote(txt)):null;
    badge.textContent=(showMidi?m!==null:false)?`MIDI ${m}`:'';
  }
}

let _svgId=0;
function svg_open(w,h,bg){const id='vp'+(++_svgId);return`<svg width="${w}" height="${h}" viewBox="0 0 ${w} ${h}" xmlns="http://www.w3.org/2000/svg" font-family="'Nunito','Segoe UI','Helvetica Neue',Arial,sans-serif" overflow="hidden"><defs><clipPath id="${id}"><rect width="${w}" height="${h}"/></clipPath></defs><rect width="${w}" height="${h}" fill="${bg||'white'}"/><g clip-path="url(#${id})">`;}
function svg_close(){return'</g></svg>';}
function tx(x,y,text,size,fill,weight,anchor){return`<text x="${x}" y="${y}" font-size="${size}" fill="${fill}" font-weight="${weight||'normal'}" text-anchor="${anchor||'start'}">${text}</text>`;}
function esc(s){if(!s)return'';return String(s).replace(/"/g,'&quot;').replace(/</g,'&lt;').replace(/>/g,'&gt;');}
function escSVG(s){if(!s)return'';return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}
function escHTML(s){if(!s)return'';const d=document.createElement('div');d.appendChild(document.createTextNode(String(s)));return d.innerHTML;}
function fmtDate(d){if(!d)return new Date().toLocaleDateString('fr-FR');try{return new Date(d).toLocaleDateString('fr-FR');}catch(e){return d;}}
function dn(note){return escSVG(displayNote(note));}  // display note in current lang
/* Render a button value (single note OR "N1/N2/N3") as SVG text centred at (cx,baseY).
   For multi-note values the notes are stacked vertically (tspan lines). */
function svgBtnVal(v,cx,baseY,color){
  if(!v)return'';
  const disp=displayNote(v);
  if(!disp.includes('/')){
    const fs=disp.length>8?5.5:disp.length>5?6.8:8;
    return tx(cx,baseY,escSVG(disp),fs,color,'bold','middle');
  }
  const parts=disp.split('/').map(p=>p.trim()).filter(Boolean);
  const lineH=6.5, fs=5.5;
  const startY=baseY-((parts.length-1)*lineH)/2;
  let out=`<text font-size="${fs}" fill="${color}" font-weight="bold" text-anchor="middle">`;
  parts.forEach((p,i)=>out+=`<tspan x="${cx}" y="${startY+i*lineH}">${escSVG(p)}</tspan>`);
  return out+'</text>';
}

/* ─── Bellows SVG ─── */
function bellowsSVG(x,y,w,h,folds,dark){
  const gold=dark?'#c9a227':'#8b6914',bg=dark?'#0a0800':'#fff8e6';
  let s=`<rect x="${x}" y="${y}" width="${w}" height="${h}" fill="${bg}" rx="2"/>`;
  const fh=h/folds;
  for(let i=0;i<folds;i++){
    const fy1=y+i*fh,fy2=fy1+fh,mid=fy1+fh/2;
    s+=`<line x1="${x}" y1="${fy1}" x2="${x+w}" y2="${fy1}" stroke="${gold}" stroke-width=".6" opacity=".7"/>`;
    s+=`<line x1="${x}" y1="${fy1}" x2="${x+w/2}" y2="${mid}" stroke="${gold}" stroke-width=".9"/>`;
    s+=`<line x1="${x+w/2}" y1="${mid}" x2="${x+w}" y2="${fy1}" stroke="${gold}" stroke-width=".9"/>`;
    if(i<folds-1){
      s+=`<line x1="${x}" y1="${fy2}" x2="${x+w/2}" y2="${mid}" stroke="${gold}" stroke-width=".6" opacity=".5"/>`;
      s+=`<line x1="${x+w/2}" y1="${mid}" x2="${x+w}" y2="${fy2}" stroke="${gold}" stroke-width=".6" opacity=".5"/>`;
    }
  }
  s+=`<rect x="${x}" y="${y}" width="${w}" height="${h}" fill="none" stroke="${gold}" stroke-width="1.5" rx="2"/>`;
  return s;
}


/* ╔══════════════════════════════════════════════════════╗
   ║  SVG KEYBOARD — CLIENT (light preview + dark PDF)    ║
   ╚══════════════════════════════════════════════════════╝ */
function buildSVG_Droite(data,dark){
  const R=24,HGAP=60,VGAP=72,LPAD=186,MARG=14,HEAD=80,FOOT=34;
  const gold=dark?'#c9a227':'#4338CA';
  const bg=dark?'#111111':'#ffffff';
  const pushBg=dark?'#0d1f0d':'#EEF2FF',pullBg=dark?'#1f1000':'#FFFBEB';
  const pushTx=dark?'#6fcf5a':'#3730A3',pullTx=dark?'#e8b840':'#92400E';
  const subtl=dark?'#888866':'#94A3B8',stroke=dark?'#c9a227':'#6366F1';
  const band1=dark?'#1a1a1a':'#FAFAFA',band2=dark?'#161616':'#F5F3FF';
  const numTx=dark?'rgba(200,162,39,.3)':'rgba(99,102,241,.2)';

  const maxBtns=Math.max(...data.droite.map(r=>r.boutons.length),1);
  const allOffs=data.droite.map(r=>r.offset||0);
  const maxOff=Math.max(...allOffs,0);
  const minOff=Math.min(...allOffs,0);
  const leftShift=-minOff*(HGAP/2);
  const BASE=LPAD+leftShift;
  const W=BASE+maxBtns*HGAP+maxOff*(HGAP/2)+(data.droite.length>1?HGAP/2:0)+MARG;
  const H=MARG+HEAD+data.droite.length*VGAP+FOOT;

  let s=svg_open(W,H,bg);

  // Orientation arrows — horizontal across the top
  const arrowY=MARG+HEAD-10;
  const ax1=BASE,ax2=W-MARG-2;
  const arrowMid=(ax1+ax2)/2;
  s+=`<line x1="${ax1+2}" y1="${arrowY}" x2="${ax2-2}" y2="${arrowY}" stroke="${gold}" stroke-width="1" opacity=".5"/>`;
  s+=`<polygon points="${ax1+2},${arrowY} ${ax1+10},${arrowY-4} ${ax1+10},${arrowY+4}" fill="${gold}" opacity=".7"/>`;
  s+=`<polygon points="${ax2-2},${arrowY} ${ax2-10},${arrowY-4} ${ax2-10},${arrowY+4}" fill="${gold}" opacity=".7"/>`;
  s+=tx(ax1+14,arrowY+4,t('svgLow'),8,gold,'bold','start');
  s+=tx(ax2-14,arrowY+4,t('svgHigh'),8,gold,'bold','end');

  // Title
  s+=tx(BASE+(maxBtns*HGAP)/2,MARG+16,escSVG(data.nomInstrument),16,gold,'bold','middle');
  if(data.nomJoueur)s+=tx(BASE+(maxBtns*HGAP)/2,MARG+32,`${t('svgClient')} : ${escSVG(data.nomJoueur)}`,10,subtl,'normal','middle');
  s+=tx(BASE+(maxBtns*HGAP)/2,MARG+47,`${t('svgLuthier')} : ${escSVG(data.nomLuthier)}  ·  ${fmtDate(data.date)}`,9,subtl,'normal','middle');

  // Legend – split circle (push=top half, pull=bottom half)
  const lx=BASE,ly=MARG+62;
  const lr=10,lcx=lx+lr,lcy=ly-4;
  s+=`<circle cx="${lcx}" cy="${lcy}" r="${lr}" fill="${bg}" stroke="${stroke}" stroke-width="1.2"/>`;
  s+=`<path d="M${lcx-lr},${lcy} A${lr},${lr} 0 0,1 ${lcx+lr},${lcy}" fill="${pushBg}"/>`;
  s+=`<path d="M${lcx-lr},${lcy} A${lr},${lr} 0 0,0 ${lcx+lr},${lcy}" fill="${pullBg}"/>`;
  s+=`<circle cx="${lcx}" cy="${lcy}" r="${lr}" fill="none" stroke="${stroke}" stroke-width="1.4"/>`;
  s+=`<line x1="${lcx-lr}" y1="${lcy}" x2="${lcx+lr}" y2="${lcy}" stroke="${stroke}" stroke-width=".5" opacity=".4"/>`;
  s+=tx(lcx+lr+5,lcy-1,t('svgPush'),8,dark?'#ccc':'#333','normal','start');
  s+=tx(lcx+lr+5,lcy+9,t('svgPull'),8,dark?'#ccc':'#333','normal','start');

  // Rows
  data.droite.forEach((row,ri)=>{
    const cy=MARG+HEAD+ri*VGAP+VGAP/2;
    const stagger=(ri%2===1)?HGAP/2:0;
    const alignOff=(row.offset||0)*(HGAP/2);
    s+=`<rect x="${BASE+alignOff-4}" y="${cy-R-6}" width="${row.boutons.length*HGAP+stagger+8}" height="${(R+6)*2}" rx="4" fill="${ri%2===0?band1:band2}" opacity=".7"/>`;
    s+=tx(BASE+alignOff-8,cy+4,escSVG(row.nom),9.5,gold,'bold','end');
    row.boutons.forEach((btn,bi)=>{
      const cx=BASE+alignOff+stagger+bi*HGAP;
      s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="${bg}" stroke="${bg==='#ffffff'?'#bbb':'#333'}" stroke-width="1.2"/>`;
      s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,1 ${cx+R},${cy}" fill="${pushBg}"/>`;
      s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,0 ${cx+R},${cy}" fill="${pullBg}"/>`;
      s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="none" stroke="${stroke}" stroke-width="1.4"/>`;
      s+=`<line x1="${cx-R}" y1="${cy}" x2="${cx+R}" y2="${cy}" stroke="${stroke}" stroke-width=".5" opacity=".4"/>`;
      s+=tx(cx,cy+3.5,row.boutons.length-bi,7.5,numTx,'normal','middle');
      if(btn.p){const fs=btn.p.length>6?7:8.5;s+=tx(cx,cy-9,dn(btn.p),fs,pushTx,'bold','middle');}
      if(btn.t){const fs=btn.t.length>6?7:8.5;s+=tx(cx,cy+19,dn(btn.t),fs,pullTx,'bold','middle');}
      if(data.showMidi){
        const mp=noteToMidi(btn.p),mt=noteToMidi(btn.t);
        if(mp!==null)s+=tx(cx+R-2,cy-R+6,mp,5.5,'rgba(100,140,220,.8)','normal','end');
        if(mt!==null)s+=tx(cx+R-2,cy+R-2,mt,5.5,'rgba(100,140,220,.8)','normal','end');
      }
    });
  });

  const fy=MARG+HEAD+data.droite.length*VGAP+14;
  s+=`<line x1="${BASE}" y1="${fy}" x2="${W-MARG-44}" y2="${fy}" stroke="${dark?'#333':'#eee'}" stroke-width="1"/>`;
  s+=tx(BASE,fy+14,`${escSVG(data.nomLuthier)} — ewendaviau.com`,7.5,dark?'#444':'#bbb','normal','start');
  s+=svg_close();return s;
}

function buildSVG_Gauche(data,dark){
  const R=24,HGAP=62,VGAP=70,LPAD=140,MARG=12,HEAD=32,FOOT=20;
  const gold=dark?'#c9a227':'#4338CA';
  const bg=dark?'#111':'#fff';
  const pushBg=dark?'#0d1f0d':'#EEF2FF',pullBg=dark?'#1f1000':'#FFFBEB';
  const pushTx=dark?'#6fcf5a':'#3730A3',pullTx=dark?'#e8b840':'#92400E';
  const stroke=dark?'#c9a227':'#6366F1';
  const band1=dark?'#1a1a1a':'#FAFAFA',band2=dark?'#161616':'#F5F3FF';
  const numTx=dark?'rgba(200,162,39,.3)':'rgba(99,102,241,.2)';
  const pushTx2=dark?'rgba(111,207,90,.55)':'rgba(55,48,163,.5)';
  const pullTx2=dark?'rgba(232,184,64,.55)':'rgba(146,64,14,.5)';

  // Paired mode: B1 A1 B2 A2 / B3 A3 B4 A4 / … (PPROW pairs per display row)
  if(data.lhDisplayMode==='paired'){
    const PPROW=Math.max(1,data.lhPairsPerRow||2); // pairs per display row
    const nPairs=Math.max(data.basses.length,data.accords.length);
    const nDispRows=Math.ceil(nPairs/PPROW);
    const nCols=PPROW*2;
    const bassLbl=dark?'#5aa0d0':'#1D4ED8';
    const accLbl=dark?'#d09040':'#B45309';
    const gridCol=dark?'#3a3a3a':'#DDD6FE';
    const W=LPAD+nCols*HGAP+MARG;
    const H=MARG+HEAD+nDispRows*VGAP+FOOT;
    let s=svg_open(W,H,bg);
    s+=tx(W/2,MARG+16,`${t('svgLH')} — ${escSVG(data.nomInstrument)}`,12,gold,'bold','middle');
    for(let r=0;r<nDispRows;r++){
      const cy=MARG+HEAD+r*VGAP+VGAP/2;
      s+=`<rect x="${LPAD-4}" y="${cy-R-6}" width="${nCols*HGAP+8}" height="${(R+6)*2}" rx="4" fill="${r%2===0?band1:band2}" opacity=".7"/>`;
      if(r>0){const yh=MARG+HEAD+r*VGAP;s+=`<line x1="${LPAD-4}" y1="${yh}" x2="${W-MARG}" y2="${yh}" stroke="${gridCol}" stroke-width="1"/>`;}
      for(let q=0;q<PPROW;q++){
        const pi=r*PPROW+q;
        if(pi>=nPairs)break;
        const bCol=data.lhBassFirst?q*2:q*2+1,aCol=data.lhBassFirst?q*2+1:q*2;
        [[data.basses[pi],'B',bCol,bassLbl,data.nbVoixBasses||2,data.basses2],[data.accords[pi],'A',aCol,accLbl,data.nbVoixAccords||2,data.accords2]].forEach(([btn,lbl,col,lblCol,voix,arr2])=>{
          if(!btn)return;
          const cx=LPAD+col*HGAP;
          const btn2=(voix>1?(arr2?arr2.length>pi:false):false)?arr2[pi]:null;
          s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="${bg}" stroke="${bg==='#fff'?'#bbb':'#333'}" stroke-width="1.2"/>`;
          s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,1 ${cx+R},${cy}" fill="${pushBg}"/>`;
          s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,0 ${cx+R},${cy}" fill="${pullBg}"/>`;
          s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="none" stroke="${stroke}" stroke-width="1.4"/>`;
          s+=`<line x1="${cx-R}" y1="${cy}" x2="${cx+R}" y2="${cy}" stroke="${stroke}" stroke-width=".5" opacity=".4"/>`;
          s+=tx(cx,cy+3.5,pi+1,7.5,numTx,'normal','middle');
          s+=tx(cx-R+5,cy-R+8,lbl,7,lblCol,'bold','start');
          const btn2HasNotes=(btn2?(btn2.p||btn2.t):false);
          if(btn2HasNotes){
            if(btn.p)s+=svgBtnVal(btn.p,cx,cy-17,pushTx);
            if(btn2.p)s+=svgBtnVal(btn2.p,cx,cy-7,pushTx2);
            if(btn.t)s+=svgBtnVal(btn.t,cx,cy+9,pullTx);
            if(btn2.t)s+=svgBtnVal(btn2.t,cx,cy+19,pullTx2);
          }else{
            if(btn.p)s+=svgBtnVal(btn.p,cx,cy-9,pushTx);
            if(btn.t)s+=svgBtnVal(btn.t,cx,cy+19,pullTx);
          }
          if(data.showMidi){
            const mp=noteToMidi(btn.p),mt=noteToMidi(btn.t);
            if(mp!==null)s+=tx(cx+R-2,cy-R+6,mp,5.5,'rgba(100,140,220,.8)','normal','end');
            if(mt!==null)s+=tx(cx+R-2,cy+R-2,mt,5.5,'rgba(100,140,220,.8)','normal','end');
          }
        });
      }
    }
    s+=svg_close();return s;
  }

  // Separated mode: multi-row basses/accords via lhRows
  const lhRows=(data.lhRows?data.lhRows.length:false)?data.lhRows:[{nb:data.basses.length||1,offset:0}];
  const secRows=[];
  let bassIdx=0,accordIdx=0;
  const vA=data.nbVoixAccords||2,vB=data.nbVoixBasses||2;
  const bassFirst=data.lhBassFirst||false;
  // Build accords sections
  const accordSecs=[];
  lhRows.forEach((row,ri)=>{
    accordSecs.push({nom:lhRows.length===1?t('accords'):(ri===0?t('accords'):`  ↳ +${accordIdx}`),items:data.accords.slice(accordIdx,accordIdx+row.nb),items2:vA>1?(data.accords2||[]).slice(accordIdx,accordIdx+row.nb):null,offset:row.offset||0,type:'A',voix:vA});
    accordIdx+=row.nb;
  });
  // Build basses sections
  const basseSecs=[];
  lhRows.forEach((row,ri)=>{
    basseSecs.push({nom:lhRows.length===1?t('basses'):(ri===0?t('basses'):`  ↳ +${bassIdx}`),items:data.basses.slice(bassIdx,bassIdx+row.nb),items2:vB>1?(data.basses2||[]).slice(bassIdx,bassIdx+row.nb):null,offset:row.offset||0,type:'B',voix:vB});
    bassIdx+=row.nb;
  });
  // Order: bassFirst → basses then accords, else accords then basses
  if(bassFirst){basseSecs.forEach(s=>secRows.push(s));accordSecs.forEach(s=>secRows.push(s));}
  else{accordSecs.forEach(s=>secRows.push(s));basseSecs.forEach(s=>secRows.push(s));}
  if(data.terzetto?data.terzetto.length:false)secRows.push({nom:t('terzetto'),items:data.terzetto,items2:null,offset:0,type:'T',voix:null});

  const maxItems=Math.max(...secRows.map(sc=>sc.items.length),1);
  const maxSecOff=Math.max(...secRows.map(sc=>sc.offset||0),0);
  const W=LPAD+maxItems*HGAP+maxSecOff*(HGAP/2)+MARG;
  const H=MARG+HEAD+secRows.length*VGAP+FOOT;

  let s=svg_open(W,H,bg);
  s+=tx(W/2,MARG+16,`${t('svgLH')} — ${escSVG(data.nomInstrument)}`,12,gold,'bold','middle');
  // Pass 1: alternating row bands + row labels
  secRows.forEach((sec,si)=>{
    const off=(sec.offset||0)*(HGAP/2);
    const cy=MARG+HEAD+si*VGAP+VGAP/2;
    if(sec.items.length)
      s+=`<rect x="${LPAD-4+off}" y="${cy-R-6}" width="${sec.items.length*HGAP+8}" height="${(R+6)*2}" rx="4" fill="${si%2===0?band1:band2}" opacity=".7"/>`;
    s+=tx(LPAD-R-10,cy,escSVG(sec.nom),10,gold,'bold','end');
    if(sec.voix!==null?sec.voix!==undefined:false)s+=tx(LPAD-R-10,cy+11,`${sec.voix}v`,7.5,dark?'rgba(200,162,39,.6)':'rgba(45,90,39,.55)','normal','end');
  });
  // Pass 2: table grid (column + row separator lines, outer border, column numbers)
  if(secRows.length>0?maxItems>0:false){
    const gTop=MARG+HEAD,gBot=MARG+HEAD+secRows.length*VGAP;
    const gLeft=LPAD-HGAP/2,gRight=LPAD+maxItems*HGAP-HGAP/2;
    const gridCol=dark?'#3a3a3a':'#DDD6FE';
    s+=`<rect x="${gLeft}" y="${gTop}" width="${gRight-gLeft}" height="${gBot-gTop}" fill="none" stroke="${dark?'#555':'#C7D2FE'}" stroke-width="1.2" rx="2"/>`;
    for(let ci=1;ci<maxItems;ci++){const xv=LPAD+ci*HGAP-HGAP/2;s+=`<line x1="${xv}" y1="${gTop}" x2="${xv}" y2="${gBot}" stroke="${gridCol}" stroke-width="1"/>`;}
    for(let ri=1;ri<secRows.length;ri++){const yh=gTop+ri*VGAP;s+=`<line x1="${gLeft}" y1="${yh}" x2="${gRight}" y2="${yh}" stroke="${gridCol}" stroke-width="1"/>`;}
    for(let ci=0;ci<maxItems;ci++){s+=tx(LPAD+ci*HGAP,gTop-5,ci+1,7,dark?'rgba(200,162,39,.5)':'rgba(99,102,241,.4)','normal','middle');}
  }
  // Pass 3: buttons on top of grid
  secRows.forEach((sec,si)=>{
    const off=(sec.offset||0)*(HGAP/2);
    const cy=MARG+HEAD+si*VGAP+VGAP/2;
    const has2=(sec.items2?sec.items2.length>0:false);
    sec.items.forEach((btn,bi)=>{
      const cx=LPAD+bi*HGAP+off;
      const btn2=has2?(sec.items2[bi]||null):null;
      s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="${bg}" stroke="${bg==='#fff'?'#bbb':'#333'}" stroke-width="1.2"/>`;
      s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,1 ${cx+R},${cy}" fill="${pushBg}"/>`;
      s+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,0 ${cx+R},${cy}" fill="${pullBg}"/>`;
      s+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="none" stroke="${stroke}" stroke-width="1.4"/>`;
      s+=`<line x1="${cx-R}" y1="${cy}" x2="${cx+R}" y2="${cy}" stroke="${stroke}" stroke-width=".5" opacity=".4"/>`;
      s+=tx(cx,cy+3.5,bi+1,7.5,numTx,'normal','middle');
      const btn2HasNotes=(has2?(btn2?(btn2.p||btn2.t):false):false);
      if(btn2HasNotes){
        // 2-voice layout: V1 outer, V2 inner (closer to center divider)
        if(btn.p)s+=svgBtnVal(btn.p,cx,cy-17,pushTx);
        if(btn2.p)s+=svgBtnVal(btn2.p,cx,cy-7,pushTx2);
        if(btn.t)s+=svgBtnVal(btn.t,cx,cy+9,pullTx);
        if(btn2.t)s+=svgBtnVal(btn2.t,cx,cy+19,pullTx2);
      } else {
        if(btn.p)s+=svgBtnVal(btn.p,cx,cy-9,pushTx);
        if(btn.t)s+=svgBtnVal(btn.t,cx,cy+19,pullTx);
      }
      if(data.showMidi){
        const mp=noteToMidi(btn.p),mt=noteToMidi(btn.t);
        if(mp!==null)s+=tx(cx+R-2,cy-R+6,mp,5.5,'rgba(100,140,220,.8)','normal','end');
        if(mt!==null)s+=tx(cx+R-2,cy+R-2,mt,5.5,'rgba(100,140,220,.8)','normal','end');
      }
    });
  });
  s+=svg_close();return s;
}

/* ── Combined landscape SVG: Right Hand + Left Hand side-by-side ── */
function buildSVG_Combined(data,dark){
  const svgD=buildSVG_Droite(data,dark);
  const svgG=buildSVG_Gauche(data,dark);
  const wD=+(svgD.match(/\bwidth="(\d+(?:\.\d+)?)"/)||[,800])[1];
  const hD=+(svgD.match(/\bheight="(\d+(?:\.\d+)?)"/)||[,600])[1];
  const wG=+(svgG.match(/\bwidth="(\d+(?:\.\d+)?)"/)||[,800])[1];
  const hG=+(svgG.match(/\bheight="(\d+(?:\.\d+)?)"/)||[,600])[1];
  const GAP=28;
  const W=wD+GAP+wG;
  const H=Math.max(hD,hG);
  const bg=dark?'#111111':'#ffffff';
  const gold=dark?'#c9a227':'#2d5a27';
  // Re-root each SVG into the outer canvas by adding x/y to the outer <svg> tag
  // (nested SVG keeps its own viewBox and clipPath — fully supported per SVG spec)
  const yD=Math.round((H-hD)/2);
  const yG=Math.round((H-hG)/2);
  const innerD=svgD.replace(/^<svg\b/,`<svg x="0" y="${yD}"`);
  const innerG=svgG.replace(/^<svg\b/,`<svg x="${wD+GAP}" y="${yG}"`);
  return [
    `<svg width="${W}" height="${H}" viewBox="0 0 ${W} ${H}" xmlns="http://www.w3.org/2000/svg" font-family="Arial,Helvetica,sans-serif" overflow="hidden">`,
    `<rect width="${W}" height="${H}" fill="${bg}"/>`,
    innerD,
    `<line x1="${wD+GAP/2}" y1="10" x2="${wD+GAP/2}" y2="${H-10}" stroke="${gold}" stroke-width="1.2" opacity=".25" stroke-dasharray="5,4"/>`,
    innerG,
    `</svg>`
  ].join('');
}

/* ╔══════════════════════════════════════════════════════╗
   ║  SVG PLAN DE CLAVIER — TikZ-style vertical layout     ║
   ║  Faithfully reproduces the LaTeX/TikZ template:       ║
   ║  · \drawtitle  table (rotated "Plan de clavier" +     ║
   ║    vertical rule + info rows)                         ║
   ║  · LH columns anchored west at 60 % vertical ratio   ║
   ║  · RH columns anchored east at 60 % vertical ratio   ║
   ║  · "Haut (graves)" in italics near top-left           ║
   ║  · Buttons in vertical columns: top=graves→bot=aigus  ║
   ║  · Horizontal divider in each circle (top=poussé,     ║
   ║    bottom=tiré)                                       ║
   ╚══════════════════════════════════════════════════════╝ */
function buildSVG_Plan(data,dark){
  /* ── constants ────────────────────────────────────── */
  const R=18;                 // button radius
  const BGAP=46;              // vertical gap between buttons (center-to-center)
  const CGAP=52;              // horizontal gap between columns
  const MH=20,MT=14,MB=30;   // margins: horiz, top, bottom
  const TITLE_H=76;           // height reserved for the title table
  const BTWN=50;              // gap between LH section and RH section (bellows)

  /* ── colours ──────────────────────────────────────── */
  const gold=dark?'#c9a227':'#4338CA';
  const bg=dark?'#111':'#fff';
  const pushBg=dark?'#0d1f0d':'#EEF2FF',pullBg=dark?'#1f1000':'#FFFBEB';
  const pushTx=dark?'#6fcf5a':'#3730A3',pullTx=dark?'#e8b840':'#92400E';
  const pushTx2=dark?'rgba(111,207,90,.55)':'rgba(55,48,163,.5)';
  const pullTx2=dark?'rgba(232,184,64,.55)':'rgba(146,64,14,.5)';
  const subtl=dark?'#888866':'#94A3B8';
  const stroke=dark?'#c9a227':'#6366F1';
  const numTx=dark?'rgba(200,162,39,.3)':'rgba(99,102,241,.2)';
  const band1=dark?'#1a1a1a':'#FAFAFA',band2=dark?'#161616':'#F5F3FF';
  const circleBorder=dark?'#333':'#C7D2FE';
  const textColor=dark?'#ccc':'#1E293B';
  const ruleColor=dark?'#444':'#E2E8F0';

  /* ── draw one button circle ───────────────────────── */
  function drawBtn(cx,cy,btn,btn2){
    let o='';
    o+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="${bg}" stroke="${circleBorder}" stroke-width="1"/>`;
    o+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,1 ${cx+R},${cy}" fill="${pushBg}"/>`;
    o+=`<path d="M${cx-R},${cy} A${R},${R} 0 0,0 ${cx+R},${cy}" fill="${pullBg}"/>`;
    o+=`<circle cx="${cx}" cy="${cy}" r="${R}" fill="none" stroke="${stroke}" stroke-width="1.2"/>`;
    o+=`<line x1="${cx-R}" y1="${cy}" x2="${cx+R}" y2="${cy}" stroke="${stroke}" stroke-width=".5" opacity=".4"/>`;
    const b2ok=(btn2?(btn2.p||btn2.t):false);
    if(b2ok){
      if(btn.p) o+=svgBtnVal(btn.p,cx,cy-11,pushTx);
      if(btn2.p)o+=svgBtnVal(btn2.p,cx,cy-4,pushTx2);
      if(btn.t) o+=svgBtnVal(btn.t,cx,cy+8,pullTx);
      if(btn2.t)o+=svgBtnVal(btn2.t,cx,cy+15,pullTx2);
    }else{
      if(btn.p)o+=svgBtnVal(btn.p,cx,cy-7,pushTx);
      if(btn.t)o+=svgBtnVal(btn.t,cx,cy+15,pullTx);
    }
    return o;
  }

  /* ── RH data ──────────────────────────────────────── */
  const rhRows=data.droite;
  const nRH=rhRows.length;
  const maxRHBtns=Math.max(...rhRows.map(r=>r.boutons.length),1);

  /* ── LH data (sections: accords, basses, terzetto) ─ */
  const isPaired=data.lhDisplayMode==='paired';
  const lhSecRows=[];

  if(isPaired){
    /* Alternated mode: split nPairs into groups of lhPairsPerRow per column.
       Vertical  → each group = 1 interleaved column (B-A-B-A stacked).
       Horizontal → each group = 2 adjacent columns (B column | A column). */
    const nPairs=Math.max(data.basses.length,data.accords.length);
    const ppRow=Math.max(1,data.lhPairsPerRow||nPairs);
    const nGroups=Math.ceil(nPairs/ppRow);
    const vA=data.nbVoixAccords||2,vB=data.nbVoixBasses||2;
    if(data.lhPairedDir==='horizontal'){
      /* Horizontal: ppRow = number of columns side by side.
         Column c holds: for each group g → B_{g*ppRow+c}, A_{g*ppRow+c} stacked vertically.
         Visual result (ppRow=3, total=6):
           Col0   Col1   Col2
           B1     B2     B3
           A1     A2     A3
           B4     B5     B6
           A4     A5     A6  */
      const nGroups=Math.ceil(nPairs/ppRow);
      for(let c=0;c<ppRow;c++){
        const intItems=[],intItems2=[],intTypes=[],intIndices=[];
        for(let g=0;g<nGroups;g++){
          const i=g*ppRow+c;
          if(i>=nPairs)break;
          const seq=data.lhBassFirst
            ?[['B',data.basses,data.basses2,vB],['A',data.accords,data.accords2,vA]]
            :[['A',data.accords,data.accords2,vA],['B',data.basses,data.basses2,vB]];
          seq.forEach(([typ,src,src2,nv])=>{
            if(src[i]){intItems.push(src[i]);intItems2.push(nv>1?src2?src2[i]||null:null:null);intTypes.push(typ);intIndices.push(i+1);}
          });
        }
        if(intItems.length){
          const lbl=ppRow===1?`${t('basses')} / ${t('accords')}`:`${t('basses')}/${t('accords')} ${c+1}`;
          lhSecRows.push({nom:lbl,items:intItems,items2:intItems2,offset:0,interleavedTypes:intTypes,interleavedIndices:intIndices});
        }
      }
    }else{
      /* Vertical (default): each group → 1 interleaved column (B-A-B-A) */
      for(let g=0;g<nGroups;g++){
        const s0=g*ppRow,s1=Math.min(s0+ppRow,nPairs);
        const intItems=[],intItems2=[],intTypes=[],intIndices=[];
        for(let i=s0;i<s1;i++){
          const seq=data.lhBassFirst
            ?[['B',data.basses,data.basses2,vB],['A',data.accords,data.accords2,vA]]
            :[['A',data.accords,data.accords2,vA],['B',data.basses,data.basses2,vB]];
          seq.forEach(([typ,src,src2,nv])=>{
            if(src[i]){intItems.push(src[i]);intItems2.push(nv>1?src2?src2[i]||null:null:null);intTypes.push(typ);intIndices.push(i+1);}
          });
        }
        const lbl=nGroups===1?`${t('basses')} / ${t('accords')}`:`${t('basses')}/${t('accords')} ${g+1}`;
        lhSecRows.push({nom:lbl,items:intItems,items2:intItems2,offset:0,interleavedTypes:intTypes,interleavedIndices:intIndices});
      }
    }
    if(data.terzetto?data.terzetto.length:false)lhSecRows.push({nom:t('terzetto'),items:data.terzetto,items2:null,offset:0});
  }else{
    /* Separated mode: distinct column per section, respects lhRows multi-row config */
    const lhRowCfg=(data.lhRows?data.lhRows.length:false)?data.lhRows:[{nb:data.basses.length||1,offset:0}];
    const accSecs=[],basSecs=[];
    let bIdx=0,aIdx=0;
    const vA=data.nbVoixAccords||2,vB=data.nbVoixBasses||2;
    lhRowCfg.forEach((row,ri)=>{
      accSecs.push({nom:lhRowCfg.length===1?t('accords'):(ri===0?t('accords'):`↳+${aIdx}`),items:data.accords.slice(aIdx,aIdx+row.nb),items2:vA>1?(data.accords2||[]).slice(aIdx,aIdx+row.nb):null,offset:row.offset||0});
      aIdx+=row.nb;
      basSecs.push({nom:lhRowCfg.length===1?t('basses'):(ri===0?t('basses'):`↳+${bIdx}`),items:data.basses.slice(bIdx,bIdx+row.nb),items2:vB>1?(data.basses2||[]).slice(bIdx,bIdx+row.nb):null,offset:row.offset||0});
      bIdx+=row.nb;
    });
    if(data.lhBassFirst){ basSecs.forEach(s=>lhSecRows.push(s)); accSecs.forEach(s=>lhSecRows.push(s)); }
    else                { accSecs.forEach(s=>lhSecRows.push(s)); basSecs.forEach(s=>lhSecRows.push(s)); }
    if(data.terzetto?data.terzetto.length:false)lhSecRows.push({nom:t('terzetto'),items:data.terzetto,items2:null,offset:0});
  }

  const nLH=lhSecRows.length;
  const maxLHBtns=nLH>0?Math.max(...lhSecRows.map(sc=>sc.items.length),0):0;
  const maxBtns=Math.max(maxRHBtns,maxLHBtns,1);

  /* ── account for row offsets in height ──────────────── */
  const rhOffsets=rhRows.map(r=>r.offset||0);
  const rhMaxOff=Math.max(...rhOffsets,0);
  const rhMinOff=Math.min(...rhOffsets,0);
  const rhTopShift=-rhMinOff*(BGAP/2);   // extra space at top for negative offsets
  const rhBottomExtra=rhMaxOff*(BGAP/2); // extra space at bottom for positive offsets
  const lhOffsets=lhSecRows.map(s=>s.offset||0);
  const lhMaxOff=lhOffsets.length?Math.max(...lhOffsets,0):0;
  const lhMinOff=lhOffsets.length?Math.min(...lhOffsets,0):0;
  const lhTopShift=-lhMinOff*(BGAP/2);
  const lhBottomExtra=lhMaxOff*(BGAP/2);
  const HEADER_GAP=120;  // space above grid for vertical column labels

  /* ── layout geometry ──────────────────────────────── */
  const lhW=nLH>0?nLH*CGAP:0;
  const rhW=nRH*CGAP;
  const contentW=lhW+(nLH>0?BTWN:0)+rhW;
  const W=Math.max(contentW+2*MH,460);
  const extraOff=Math.max(rhTopShift+rhBottomExtra,lhTopShift+lhBottomExtra);
  const bodyH=maxBtns*BGAP+extraOff+HEADER_GAP+20;
  const H=MT+TITLE_H+bodyH+MB;

  /* positions — matching TikZ template exactly:
     · \input{left}  (anchor=west)  = RH (main droite) on LEFT side
     · \input{right} (anchor=east)  = LH (main gauche) on RIGHT side
     · both centred at 60 % vertical depth (TikZ .6 ratio)             */
  const contentTop=MT+TITLE_H;
  const contentBot=H-MB;
  const gridCenterY=contentTop+(contentBot-contentTop)*0.55;
  const gridTop=Math.round(gridCenterY-bodyH/2)+HEADER_GAP;
  /* gridTop now leaves HEADER_GAP above for rotated labels */

  /* RH on LEFT: outermost row (Ré/ext.) far left, innermost (Sol/int.) closest to bellows */
  const rhX0=MH+CGAP/2;
  /* LH on RIGHT: innermost (accords) closest to bellows, outermost (basses) far right */
  const lhX0=nLH>0? rhX0+rhW+BTWN : rhX0+rhW;
  /* bellows separator halfway between RH right edge and LH left edge */
  const sepX=nLH>0?(rhX0+(nRH-1)*CGAP+R+(lhX0-R))/2 : rhX0+rhW+BTWN/2;

  let s=svg_open(W,H,bg);

  /* ── Thin accent bar at top ──────────────────────── */
  s+=`<rect x="0" y="0" width="${W}" height="3" fill="${gold}"/>`;

  /* ═══════════════════════════════════════════════════
     TITLE TABLE — reproduces TikZ \drawtitle
     ═══════════════════════════════════════════════════ */
  const tblX=MH, tblY=MT;
  const rotW=22;                   // width of rotated-title column
  const tblInfoX=tblX+rotW+6;     // left edge of info rows
  const rowH=16;                   // row height
  const tblRows=4;
  const tblH=tblRows*rowH;

  // accent background behind the "Plan de clavier" rotated title column
  s+=`<rect x="${tblX}" y="${tblY}" width="${rotW}" height="${tblH}" fill="${gold}" opacity=".08" rx="3"/>`;
  // outer border of the title table
  s+=`<rect x="${tblX}" y="${tblY}" width="${rotW+244}" height="${tblH}" fill="none" stroke="${ruleColor}" stroke-width="1" rx="3"/>`;
  // vertical rule between rotated title and info columns
  s+=`<line x1="${tblX+rotW}" y1="${tblY}" x2="${tblX+rotW}" y2="${tblY+tblH}" stroke="${ruleColor}" stroke-width="1"/>`;
  // horizontal rules between info rows
  for(let i=1;i<tblRows;i++)
    s+=`<line x1="${tblX+rotW}" y1="${tblY+i*rowH}" x2="${tblX+rotW+244}" y2="${tblY+i*rowH}" stroke="${ruleColor}" stroke-width=".5"/>`;

  // rotated "Plan de clavier" (reads bottom-to-top, matching \begin{turn}{90})
  const rotCx=tblX+rotW/2, rotCy=tblY+tblH/2;
  s+=`<text x="${rotCx}" y="${rotCy}" text-anchor="middle" font-size="13" fill="${gold}" font-weight="bold" transform="rotate(-90,${rotCx},${rotCy})">${escSVG(t('svgPlan'))}</text>`;

  // Row 1 — system / instrument name
  s+=`<text x="${tblInfoX}" y="${tblY+rowH/2+4}" font-size="12" fill="${gold}" font-weight="bold">${escSVG(data.nomInstrument||'Accordéon Diatonique')}</text>`;
  // Row 2 — client
  const r2=data.nomJoueur?`${t('svgClient')} : ${data.nomJoueur}`:'';
  s+=`<text x="${tblInfoX}" y="${tblY+rowH+rowH/2+3}" font-size="8.5" fill="${textColor}">${escSVG(r2)}</text>`;
  // Row 3 — model description (show actual basses + accords count)
  const nBtnsRH=data.droite.reduce((a,r)=>a+r.boutons.length,0);
  const nBasses=data.basses?data.basses.length:0;
  const nAccords=data.accords?data.accords.length:0;
  const r3=`${nRH} ${nRH>1?'rangées':'rangée'} (${nBtnsRH} boutons) · ${nBasses} basses · ${nAccords} accords`;
  s+=`<text x="${tblInfoX}" y="${tblY+2*rowH+rowH/2+3}" font-size="8" fill="${textColor}">${escSVG(r3)}</text>`;
  // Row 4 — luthier + date
  s+=`<text x="${tblInfoX}" y="${tblY+3*rowH+rowH/2+3}" font-size="7.5" fill="${subtl}">${escSVG(data.nomLuthier)}  ·  ${escSVG(fmtDate(data.date))}</text>`;

  /* ── "Haut (graves)" in italics near top-left ─────── */
  s+=`<text x="${rhX0-R-6}" y="${gridTop-HEADER_GAP+4}" font-size="8.5" fill="${subtl}" font-style="italic">Haut (graves)</text>`;

  /* ── push/pull legend (small circle, top-right) ───── */
  const legR=7,legX=W-MH-legR-2,legY=tblY+tblH/2;
  s+=`<circle cx="${legX}" cy="${legY}" r="${legR}" fill="${bg}" stroke="${stroke}" stroke-width="1"/>`;
  s+=`<path d="M${legX-legR},${legY} A${legR},${legR} 0 0,1 ${legX+legR},${legY}" fill="${pushBg}"/>`;
  s+=`<path d="M${legX-legR},${legY} A${legR},${legR} 0 0,0 ${legX+legR},${legY}" fill="${pullBg}"/>`;
  s+=`<circle cx="${legX}" cy="${legY}" r="${legR}" fill="none" stroke="${stroke}" stroke-width="1"/>`;
  s+=`<line x1="${legX-legR}" y1="${legY}" x2="${legX+legR}" y2="${legY}" stroke="${stroke}" stroke-width=".4" opacity=".4"/>`;
  s+=tx(legX-legR-3,legY-1,t('svgPush').replace(/[↓↑]/g,'').trim(),6,textColor,'normal','end');
  s+=tx(legX-legR-3,legY+7,t('svgPull').replace(/[↓↑]/g,'').trim(),6,textColor,'normal','end');

  /* ═══════════════════════════════════════════════════
     RH COLUMNS  — anchored west (LEFT side of page)
     Matches TikZ: \node[anchor=west]{\input{left}}
     Row 0 (R1/Sol/int.) on far left, last row on right
     ═══════════════════════════════════════════════════ */
  s+=tx(rhX0+((nRH-1)*CGAP)/2,gridTop-HEADER_GAP+6,t('svgRH'),8.5,gold,'bold','middle');
  rhRows.forEach((row,ri)=>{
    const cx=rhX0+ri*CGAP;  // ri=0=R1 → leftmost
    const off=rhTopShift+(row.offset||0)*(BGAP/2);  // normalized: topShift ensures negative offsets stay visible
    // column label — fully vertical (-90°) above the column, text goes upward from lblY
    const lblY=gridTop-8;
    s+=`<text x="${cx}" y="${lblY}" text-anchor="start" font-size="6.5" fill="${gold}" font-weight="bold" transform="rotate(-90,${cx},${lblY})">${escSVG(row.nom)}</text>`;
    // alternating column band
    if(row.boutons.length){
      const cy0=gridTop+off+BGAP/2;
      const bandY=cy0-R-3;
      const bandH=(row.boutons.length-1)*BGAP+2*(R+3);
      s+=`<rect x="${cx-R-3}" y="${bandY}" width="${(R+3)*2}" height="${bandH}" rx="3" fill="${ri%2===0?band1:band2}" opacity=".6"/>`;
    }
    // buttons + per-column button number (bottom-right corner, with row prime suffix)
    const suffix=ri===0?'':ri===1?"'":"''";
    row.boutons.forEach((btn,bi)=>{
      const cy=gridTop+off+bi*BGAP+BGAP/2;
      s+=drawBtn(cx,cy,btn,null);
      s+=tx(cx-R-1,cy+R+2,(bi+1)+suffix,5.5,numTx,'bold','start');
    });
  });

  /* ═══════════════════════════════════════════════════
     BELLOWS SEPARATOR — vertical dashed line
     ═══════════════════════════════════════════════════ */
  if(nLH>0){
    const sepY1=gridTop-2,sepY2=gridTop+maxBtns*BGAP+extraOff+2;
    s+=`<line x1="${sepX}" y1="${sepY1}" x2="${sepX}" y2="${sepY2}" stroke="${dark?'#444':'#C7D2FE'}" stroke-width="1.5" stroke-dasharray="6,4"/>`;
    const sepMidY=(sepY1+sepY2)/2;
    s+=`<text x="${sepX}" y="${sepMidY}" text-anchor="middle" font-size="6.5" fill="${subtl}" font-weight="bold" transform="rotate(-90,${sepX},${sepMidY})">${escSVG('— '+t('svgBellows')+' —')}</text>`;
  }

  /* ═══════════════════════════════════════════════════
     LH COLUMNS  — anchored east (RIGHT side of page)
     Matches TikZ: \node[anchor=east]{\input{right}}
     Innermost sections (accords) closest to bellows = leftmost
     Outermost sections (basses) = rightmost (far right)
     LH vertically centred relative to RH block
     ═══════════════════════════════════════════════════ */
  if(nLH>0?maxLHBtns>0:false){
    // Vertical offset to centre LH relative to RH
    const rhBlockH=maxRHBtns*BGAP+rhTopShift+rhBottomExtra;
    const lhBlockH=maxLHBtns*BGAP+lhTopShift+lhBottomExtra;
    const lhGridTop=gridTop+Math.round((rhBlockH-lhBlockH)/2);

    s+=tx(lhX0+((nLH-1)*CGAP)/2,lhGridTop-HEADER_GAP+6,t('svgLH'),8.5,gold,'bold','middle');

    /* All LH sections: unified render loop.
       isPaired mode pushes a single interleaved column into lhSecRows (B-A-B-A).
       Separated mode pushes one column per section. */
    const bassLbl=dark?'#5aa0d0':'#1D4ED8';
    const accLbl=dark?'#d09040':'#B45309';
    lhSecRows.forEach((sec,si)=>{
      const cx=lhX0+si*CGAP;
      const off=lhTopShift+(sec.offset||0)*(BGAP/2);
      const has2=(sec.items2?sec.items2.some(x=>x!=null):false);
      // alternating column band
      if(sec.items.length){
        const cy0=lhGridTop+off+BGAP/2;
        const bandY=cy0-R-3;
        const bandH=(sec.items.length-1)*BGAP+2*(R+3);

      }
      // buttons + B/A markers for interleaved mode
      sec.items.forEach((btn,bi)=>{
        const cy=lhGridTop+off+bi*BGAP+BGAP/2;
        s+=drawBtn(cx,cy,btn,has2?(sec.items2[bi]||null):null);
        if(sec.interleavedTypes){
          const typ=sec.interleavedTypes[bi];
          const idx=sec.interleavedIndices?sec.interleavedIndices[bi]:'';
          s+=tx(cx-R+4,cy-R-2,typ+idx,5.5,typ==='B'?bassLbl:accLbl,'bold','start');
        }
      });
    });
    // button numbers on the far right
    const lhLastCx=lhX0+(nLH-1)*CGAP;
    for(let i=0;i<maxLHBtns;i++)
      s+=tx(lhLastCx+R+5,lhGridTop+i*BGAP+BGAP/2+3,i+1,6,numTx,'normal','start');
  }

  /* ── "↑ Aigus" centred at bottom ───────────────── */
  s+=`<text x="${W/2}" y="${H-MB+12}" font-size="8.5" fill="${textColor}" font-style="italic" text-anchor="middle">↑ Aigus</text>`;

  /* ── footer ────────────────────────────────────── */
  s+=`<line x1="${MH}" y1="${H-MB+20}" x2="${W-MH}" y2="${H-MB+20}" stroke="${ruleColor}" stroke-width="1"/>`;
  s+=tx(MH,H-MB+28,`${escSVG(data.nomLuthier)} — ewendaviau.com`,7,subtl,'normal','start');

  s+=svg_close();
  return s;
}

function shiftOctave(noteStr,octaves){
  if(!noteStr||!octaves)return noteStr;
  const m=noteStr.match(/^(.+?)(\d+)$/);
  if(!m)return noteStr;
  return m[1]+Math.max(0,parseInt(m[2])+octaves);
}
function expandVoicesForSommier(rows,voixPerRow){
  const legacyMap={N:['N'],BN:['B','N'],NH:['N','A'],BNH:['B','N','A']};
  const arr=Array.isArray(voixPerRow)?voixPerRow:(legacyMap[voixPerRow]||['N']);
  if(arr.length<=1)return rows;
  const result=[];
  for(const row of rows){
    for(const vtype of arr){
      const vd=VOICE_TYPE_DATA[vtype]||VOICE_TYPE_DATA['N'];
      result.push({

        boutons:(row.boutons||[]).map(b=>({p:shiftOctave(b.p,vd.oct),t:shiftOctave(b.t,vd.oct)})),
        offset:row.offset||0
      });
    }
  }
  return result;
}
function buildSommierSection(title,rows,dark){
  const CW=58,CH=46,MARG=12,HEAD=52,VGAP=CH+6,FOOT=20;
  // Compute LPAD dynamically from the longest row label
  // Estimate: ~5.6px per char at font-size 9 bold + 14px padding
  const maxLabelLen=Math.max(...rows.map(r=>(r.nom||r.label||'').length),0);
  const LPAD=Math.max(120,Math.ceil(maxLabelLen*5.6)+14);
  const gold=dark?'#c9a227':'#2d5a27';
  const bg=dark?'#111':'#fff';
  const pushBg=dark?'#0d1f0d':'#e8f5e3',pullBg=dark?'#1f1000':'#fff8e6';
  const pushTx=dark?'#6fcf5a':'#1a5c1a',pullTx=dark?'#e8b840':'#8b6914';
  const numTx=dark?'rgba(200,162,39,.4)':'rgba(0,0,0,.3)';
  const sep=dark?'#3a3a1a':'#cccccc';
  const showMidi=document.getElementById('show-midi')?.checked;

  function cell(cx,cy,num,pNote,tNote){
    let o='';
    o+=`<rect x="${cx}" y="${cy}" width="${CW}" height="${CH/2}" fill="${pushBg}" stroke="${dark?'#2a4a2a':'#b0ccb0'}" stroke-width=".7"/>`;
    o+=`<rect x="${cx}" y="${cy+CH/2}" width="${CW}" height="${CH/2}" fill="${pullBg}" stroke="${dark?'#4a2a00':'#ccb080'}" stroke-width=".7"/>`;
    o+=`<line x1="${cx}" y1="${cy+CH/2}" x2="${cx+CW}" y2="${cy+CH/2}" stroke="${sep}" stroke-width=".8"/>`;
    o+=tx(cx+3,cy+9,num,6.5,numTx,'normal','start');
    if(pNote){const fs=pNote.length>6?7:8.5;o+=tx(cx+CW/2,cy+16,dn(pNote),fs,pushTx,'bold','middle');}
    if(tNote){const fs=tNote.length>6?7:8.5;o+=tx(cx+CW/2,cy+CH-7,dn(tNote),fs,pullTx,'bold','middle');}
    if(showMidi){
      const mp=noteToMidi(pNote),mt=noteToMidi(tNote);
      if(mp!==null)o+=tx(cx+CW-2,cy+9,mp,5.5,'rgba(100,140,220,.8)','normal','end');
      if(mt!==null)o+=tx(cx+CW-2,cy+CH-2,mt,5.5,'rgba(100,140,220,.8)','normal','end');
    }
    return o;
  }

  const maxBtns=Math.max(...rows.map(r=>r.boutons?r.boutons.length:(r.items?r.items.length:0)),1);
  const allOffs=rows.map(r=>r.offset||0);
  const maxOff=Math.max(...allOffs,0);
  const minOff=Math.min(...allOffs,0);
  const leftShift=-minOff*(CW/2);
  const RBASE=LPAD+leftShift;
  const W=RBASE+maxBtns*CW+maxOff*(CW/2)+MARG;
  const H=MARG+HEAD+rows.length*VGAP+FOOT;

  s+=`<rect x="${RBASE}" y="${MARG+25}" width="13" height="8" rx="1" fill="${pushBg}" stroke="${dark?'#4a7a40':'#2d5a27'}" stroke-width=".8"/>`;
  s+=tx(RBASE+17,MARG+33,t('svgPush'),7.5,dark?'#ccc':'#333','normal','start');

  s+=tx(RBASE+85,MARG+33,t('svgPull'),7.5,dark?'#ccc':'#333','normal','start');
  if(showMidi)s+=tx(RBASE+135,MARG+33,'MIDI = coin ↗',7,dark?'rgba(100,140,220,.8)':'rgba(21,101,192,.6)','normal','start');

  rows.forEach((row,ri)=>{
    const rowY=MARG+HEAD+ri*VGAP;

    s+=tx(RBASE+alignOff-5,rowY+CH/2+4,escSVG(nom),9,gold,'bold','end');
    items.forEach((btn,bi)=>{s+=cell(RBASE+alignOff+bi*CW,rowY,bi+1,btn.p||'',btn.t||'');});
  });
  s+=svg_close();return s;
}

function buildLhSommierSections(data){
  const lhRows=(data.lhRows?data.lhRows.length:false)?data.lhRows:[{nb:Math.max(data.basses.length,data.accords.length)||1,offset:0}];
  const lhSections=[];
  let bIdx=0,aIdx=0;
  const vB=data.nbVoixBasses||2, vA=data.nbVoixAccords||2;
  lhRows.forEach((row,ri)=>{
    const rowLabel=lhRows.length===1?'':`${ri+1}`;
    const bSlice=data.basses?data.basses.slice(bIdx,bIdx+row.nb):[];
    const aSlice=data.accords?data.accords.slice(aIdx,aIdx+row.nb):[];
    if(bSlice.length){
      const basNom=lhRows.length===1?t('basses'):t('basses')+' '+rowLabel;
      const _bkeys=['basses','basses2','basses3'];
      for(let v=0;v<vB;v++){
        const vSlice=(data[_bkeys[v]]||data.basses||[]).slice(bIdx,bIdx+row.nb);
        lhSections.push({nom:basNom+(vB>1?` — V${v+1}`:''),items:vSlice,offset:row.offset||0});
      }
    }
    if(aSlice.length){
      const accNom=lhRows.length===1?t('accords'):t('accords')+' '+rowLabel;
      const _akeys=['accords','accords2','accords3'];
      for(let v=0;v<vA;v++){
        const vSlice=(data[_akeys[v]]||data.accords||[]).slice(aIdx,aIdx+row.nb);
        lhSections.push({nom:accNom+(vA>1?` — V${v+1}`:''),items:vSlice,offset:row.offset||0});
      }
    }
    bIdx+=row.nb;aIdx+=row.nb;
  });
  if(data.terzetto?data.terzetto.length:false)lhSections.push({nom:t('terzetto'),items:data.terzetto,offset:0});
  return lhSections;
}

/* Scale svgTgt to same natural width as svgRef (centers content via viewBox offset).
   This ensures CSS max-width:100% scales both SVGs identically → same visual text size. */
function matchSvgWidth(svgRef,svgTgt){
  const wRef=+(svgRef.match(/\bwidth="(\d+(?:\.\d+)?)"/)||[,0])[1];
  const wTgt=+(svgTgt.match(/\bwidth="(\d+(?:\.\d+)?)"/)||[,0])[1];
  const hTgt=+(svgTgt.match(/\bheight="(\d+(?:\.\d+)?)"/)||[,0])[1];
  if(wRef<=wTgt||wTgt===0)return svgTgt;
  const xOff=(wRef-wTgt)/2;
  return svgTgt.replace(/<svg [^>]+>/,
    `<svg width="${wRef}" height="${hTgt}" viewBox="${-xOff} 0 ${wRef} ${hTgt}" xmlns="http://www.w3.org/2000/svg" font-family="Arial,Helvetica,sans-serif">`);
}

function loadScript(url){
  return new Promise((res,rej)=>{
    if(document.querySelector(`script[src="${url}"]`)){res();return;}
    const s=document.createElement('script');
    s.src=url;s.onload=res;s.onerror=rej;
    document.head.appendChild(s);
  });
}
function svgToPngDataUrl(svgStr,rotAngle){
  return new Promise((res,rej)=>{
    const mw=svgStr.match(/\bwidth="(\d+(?:\.\d+)?)"/),mh=svgStr.match(/\bheight="(\d+(?:\.\d+)?)"/);
    const sw=mw?+mw[1]:800,sh=mh?+mh[1]:600;
    const SCALE=2;
    const rot=rotAngle||0;
    const cw=(rot===90||rot===-90)?sh:sw;
    const ch=(rot===90||rot===-90)?sw:sh;
    const c=document.createElement('canvas');c.width=cw*SCALE;c.height=ch*SCALE;
    const ctx=c.getContext('2d');ctx.scale(SCALE,SCALE);
    const img=new Image();
    img.onload=()=>{
      ctx.fillStyle='#fff';ctx.fillRect(0,0,cw,ch);
      ctx.save();
      ctx.beginPath();ctx.rect(0,0,cw,ch);ctx.clip();
      if(rot===90){ctx.translate(ch,0);ctx.rotate(Math.PI/2);}
      else if(rot===-90){ctx.translate(0,cw);ctx.rotate(-Math.PI/2);}
      ctx.drawImage(img,0,0,sw,sh);
      ctx.restore();
      res({dataUrl:c.toDataURL('image/jpeg',0.90),w:cw,h:ch});
      URL.revokeObjectURL(img.src);
    };
    img.onerror=e=>{URL.revokeObjectURL(img.src);rej(e);};
    // Blob URL is more reliable than data: URL for SVG clip-path rendering in all browsers
    const blob=new Blob([svgStr],{type:'image/svg+xml'});
    img.src=URL.createObjectURL(blob);
  });
}
async function downloadPDF(which){
  const JSPDF_URL='https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';
  const AT_URL='https://cdnjs.cloudflare.com/ajax/libs/jspdf-autotable/3.8.2/jspdf.plugin.autotable.min.js';
  try{ await loadScript(JSPDF_URL); await loadScript(AT_URL); }
  catch(e){ alert(t('popupAlert'));return; }

  const data=readInputs();
  const {jsPDF}=window.jspdf;
  // 'visual' PDF with plan mode: portrait A4 with single plan SVG
  // 'visual' PDF normal: portrait A4 with stacked SVGs
  // 'fabricant' PDF: portrait A4 with separate SVGs + tables
  const pdfOrientation='portrait';
  const pdf=new jsPDF({orientation:pdfOrientation,unit:'mm',format:'a4'});
  const PW=pdf.internal.pageSize.getWidth(), PH=pdf.internal.pageSize.getHeight();

  // ── Layout constants ──────────────────────────────
  const ML=15, MR=15, MT=15;
  const CW=PW-ML-MR;                 // 180 mm usable width
  const FOOTER_H=12;                  // mm reserved at bottom for footer area
  const FOOT_RULE=PH-10;             // Y of footer rule (2mm gap to content)
  const CB=PH-FOOTER_H;             // content bottom limit
  let yPos=MT;

  // ── Professional palette ──────────────────────────
  const GRN=[67,56,202];             // deep indigo — headings
  const GRN2=[99,102,241];           // indigo-500 — sub-headers / circle stroke
  const GRAY=[100,116,139];          // slate-500 — meta / captions
  const LGRE=[238,242,255];          // indigo-50 — key cells / push background
  const ALT=[245,243,255];           // violet-50 — alternate row tint
  const RULE_C=[199,210,254];        // indigo-200 — table borders / rules
  const CIRC_C=[165,180,252];        // indigo-300 — button circle border
  const PUSH_C=[55,48,163];          // push note — indigo-800
  const PULL_C=[146,64,14];          // pull note — amber-800
  const PULL_C2=[217,119,6];         // pull note 2nd voice — amber-600

  // ── Drawing helpers ───────────────────────────────
  function hRule(y,color=RULE_C,lw=0.2){
    pdf.setDrawColor(...color);pdf.setLineWidth(lw);
    pdf.line(ML,y,PW-MR,y);
  }
  function secLabel(txt,y){
    pdf.setFontSize(6.5);pdf.setFont('helvetica','bold');pdf.setTextColor(...GRAY);
    pdf.text(txt.toUpperCase(),ML,y);
  }

  // ── SVG → PDF with JPEG rendering ────────
  // forceOnePage: if true, scale down so the image fits on one page
  async function addSvgImg(svgStr,label,mmPerPx,forceOnePage){
    const {dataUrl,w,h}=await svgToPngDataUrl(svgStr,0);
    let imgW=mmPerPx?Math.min(w*mmPerPx,CW):CW;
    let scale=imgW/w;            // mm per natural pixel
    let imgH=h*scale;            // total height in mm
    if(label){
      if(yPos+8>CB){pdf.addPage();yPos=MT;}
      secLabel(label,yPos+4);yPos+=7;
    }
    // When forceOnePage, scale down so image fits in remaining space on this page
    if(forceOnePage?imgH>CB-yPos:false){
      scale=(CB-yPos)/h;
      imgH=CB-yPos;
      imgW=w*scale;
    }
    const xImg=ML+(CW-imgW)/2;  // center horizontally (computed after forceOnePage adjustment)
    if(imgH<=CB-yPos){
      pdf.addImage(dataUrl,'JPEG',xImg,yPos,imgW,imgH);
      yPos+=imgH+5;return;
    }
    // Spans multiple pages — slice using actual canvas coords
    const srcImg=await new Promise((res,rej)=>{
      const i=new Image();i.onload=()=>res(i);i.onerror=rej;i.src=dataUrl;
    });
    const pxW=srcImg.naturalWidth, canvasScale=srcImg.naturalHeight/h; // canvas pixels per SVG pixel (=SCALE=3)
    const sc=document.createElement('canvas');sc.width=pxW;
    let srcY=0,first=true;
    while(srcY<h){
      if(!first){pdf.addPage();yPos=MT;}
      const maxH=Math.min(h-srcY,(CB-yPos)/scale);
      sc.height=Math.ceil(maxH*canvasScale);
      const ctx=sc.getContext('2d');
      ctx.clearRect(0,0,pxW,sc.height);
      ctx.drawImage(srcImg,0,srcY*canvasScale,pxW,maxH*canvasScale,0,0,pxW,sc.height);
      pdf.addImage(sc.toDataURL('image/jpeg',0.90),'JPEG',xImg,yPos,imgW,maxH*scale);
      yPos+=maxH*scale+1;srcY+=maxH;first=false;
    }
    yPos+=3;
  }

  // ── Plan de clavier — dessiné nativement avec les primitives jsPDF ──
  // Texte 100 % natif PDF (sélectionnable/copiable). Reproduit le layout de buildSVG_Plan.
  function drawPlanNative(data){
    /* ─ constantes de layout (mêmes que buildSVG_Plan en pixels SVG) ─ */
    const R_S=18,BGAP_S=46,CGAP_S=52,MH_S=20,MT_S=14,MB_S=30;
    const TITLE_H_S=76,BTWN_S=50,HEADER_GAP_S=120;

    /* ─ reconstruction des colonnes MG (même logique que buildSVG_Plan) ─ */
    const lhSecs=[];
    if(data.lhDisplayMode==='paired'){
      const nP=Math.max((data.basses||[]).length,(data.accords||[]).length);
      const ppR=Math.max(1,data.lhPairsPerRow||nP);
      const nG=Math.ceil(nP/ppR);
      for(let g=0;g<nG;g++){
        const s0=g*ppR,s1=Math.min(s0+ppR,nP),its=[],its2=[],tys=[],ixs=[];
        for(let i=s0;i<s1;i++){
          const seq=data.lhBassFirst
            ?[['B',data.basses,data.basses2,data.nbVoixBasses||2],['A',data.accords,data.accords2,data.nbVoixAccords||2]]
            :[['A',data.accords,data.accords2,data.nbVoixAccords||2],['B',data.basses,data.basses2,data.nbVoixBasses||2]];
          seq.forEach(([ty,src,sr2,nv])=>{if(src?src[i]:false){its.push(src[i]);its2.push(nv>1?sr2?sr2[i]||null:null:null);tys.push(ty);ixs.push(i+1);}});
        }
        if(its.length)lhSecs.push({nom:nG===1?`${t('basses')}/${t('accords')}`:`${t('basses')}/${t('accords')} ${g+1}`,items:its,items2:its2,offset:0,iTypes:tys,iIdxs:ixs});
      }
    }else{
      const cfg=(data.lhRows?data.lhRows.length:false)?data.lhRows:[{nb:(data.basses||[]).length||1,offset:0}];
      const aS=[],bS=[];let aI=0,bI=0;
      const vA=data.nbVoixAccords||2,vB=data.nbVoixBasses||2;
      cfg.forEach((row,ri)=>{
        aS.push({nom:cfg.length===1?t('accords'):(ri===0?t('accords'):`↳+${aI}`),items:(data.accords||[]).slice(aI,aI+row.nb),items2:vA>1?(data.accords2||[]).slice(aI,aI+row.nb):null,offset:row.offset||0});aI+=row.nb;
        bS.push({nom:cfg.length===1?t('basses'):(ri===0?t('basses'):`↳+${bI}`),items:(data.basses||[]).slice(bI,bI+row.nb),items2:vB>1?(data.basses2||[]).slice(bI,bI+row.nb):null,offset:row.offset||0});bI+=row.nb;
      });
      if(data.lhBassFirst){bS.forEach(s=>lhSecs.push(s));aS.forEach(s=>lhSecs.push(s));}
      else{aS.forEach(s=>lhSecs.push(s));bS.forEach(s=>lhSecs.push(s));}
    }
    if(data.terzetto?data.terzetto.length:false)lhSecs.push({nom:t('terzetto'),items:data.terzetto,items2:null,offset:0});

    /* ─ dimensions ─ */
    const rhR=data.droite||[],nRH=rhR.length,nLH=lhSecs.length;
    const maxRH=Math.max(...rhR.map(r=>(r.boutons||[]).length),1);
    const maxLH=nLH>0?Math.max(...lhSecs.map(s=>s.items.length),0):0;
    const maxB=Math.max(maxRH,maxLH,1);
    const rhTS=Math.max(0,...rhR.map(r=>-(r.offset||0)))*(BGAP_S/2);
    const rhBE=Math.max(0,...rhR.map(r=>(r.offset||0)))*(BGAP_S/2);
    const lhTS=nLH?Math.max(0,...lhSecs.map(s=>-(s.offset||0)))*(BGAP_S/2):0;
    const lhBE=nLH?Math.max(0,...lhSecs.map(s=>(s.offset||0)))*(BGAP_S/2):0;
    const eOff=Math.max(rhTS+rhBE,lhTS+lhBE);
    const W_S=Math.max(nRH*CGAP_S+(nLH>0?BTWN_S+nLH*CGAP_S:0)+2*MH_S,460);
    const bodyH_S=maxB*BGAP_S+eOff+HEADER_GAP_S+20;
    const H_S=MT_S+TITLE_H_S+bodyH_S+MB_S;

    /* ─ mise à l'échelle pour tenir sur la page ─ */
    const availH=Math.max(CB-yPos,80);
    const S=Math.min(CW/W_S, availH/H_S);  // mm par pixel SVG
    const planW=W_S*S, planH=H_S*S;
    if(yPos+planH>CB+0.1){pdf.addPage();yPos=MT;}
    const ox=ML+(CW-planW)/2, oy=yPos;

    /* ─ convertisseurs de coordonnées ─ */
    function px(x){return ox+x*S;}
    function py(y){return oy+y*S;}
    function pr(r){return Math.max(r*S,0.05);}
    // taille de police : n unités SVG × S (mm/unité) = n*S mm ; 1pt = 0.353mm → pt = n*S/0.353
    function fs(n){return Math.max(n*S/0.353,2.5);}

    /* ─ positions géométriques (mêmes calculs que buildSVG_Plan) ─ */
    const cTop=MT_S+TITLE_H_S,cBot=H_S-MB_S;
    const gCY=cTop+(cBot-cTop)*0.55;
    const gridTop=Math.round(gCY-bodyH_S/2)+HEADER_GAP_S;
    const rhX0=MH_S+CGAP_S/2;
    const lhX0=nLH>0?rhX0+nRH*CGAP_S+BTWN_S:rhX0+nRH*CGAP_S;
    const sepX=nLH>0?(rhX0+(nRH-1)*CGAP_S+R_S+(lhX0-R_S))/2:rhX0+nRH*CGAP_S+BTWN_S/2;

    /* ─ tableau de titre ─ */
    const tX=MH_S,tY=MT_S,rotW=22,infoX=tX+rotW+6,rH=16,tH=64;
    // thin accent bar at the very top of the page
    pdf.setFillColor(...GRN);
    pdf.rect(ox,oy,planW,pr(3),'F');
    // accent background behind the rotated title column
    pdf.setFillColor(...LGRE);
    pdf.roundedRect(px(tX),py(tY),pr(rotW),pr(tH),pr(3),pr(3),'F');
    // outer border of title table
    pdf.setDrawColor(...RULE_C);pdf.setLineWidth(0.3);
    pdf.roundedRect(px(tX),py(tY),pr(rotW+244),pr(tH),pr(3),pr(3),'S');
    pdf.line(px(tX+rotW),py(tY),px(tX+rotW),py(tY+tH));
    for(let i=1;i<4;i++){pdf.setLineWidth(0.15);pdf.line(px(infoX-2),py(tY+i*rH),px(infoX+244),py(tY+i*rH));}
    // "Plan de clavier" pivoté à 90° — centré dans le bloc titre
    // jsPDF angle:90 → le texte monte depuis l'ancre; l'ancre est en bas de la colonne.
    // Pour centrer verticalement à py(tY+tH/2), on pose l'ancre à py(tY+tH/2)+tw/2.
    // N'utilise PAS align:'center' : jsPDF déplace x avant rotation, ce qui décale le centre.
    {const label=t('svgPlan');
    pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
    let planFs=fs(12.5);pdf.setFontSize(planFs);
    const maxH=tH*S;let tw=pdf.getTextWidth(label);
    if(tw>maxH){planFs=Math.max(planFs*maxH/tw,3);pdf.setFontSize(planFs);tw=pdf.getTextWidth(label);}
    pdf.text(label,px(tX+rotW/2),py(tY+tH/2)+tw/2,{angle:90});}
    // nom de l'instrument
    pdf.setFontSize(fs(11));
    pdf.text(data.nomInstrument||'Accordéon',px(infoX),py(tY+rH*0.7));
    // client
    pdf.setFontSize(fs(8));pdf.setFont('helvetica','normal');pdf.setTextColor(...GRAY);
    if(data.nomJoueur)pdf.text(`${t('svgClient')}: ${data.nomJoueur}`,px(infoX),py(tY+rH+rH*0.65));
    // description
    const nBR=rhR.reduce((a,r)=>a+(r.boutons||[]).length,0);
    pdf.text(`${nRH} rang${nRH>1?'ées':'ée'} (${nBR} btn) · ${(data.basses||[]).length} basses · ${(data.accords||[]).length} accords`,px(infoX),py(tY+2*rH+rH*0.65));
    // luthier + date
    pdf.setFontSize(fs(7.5));
    pdf.text(`${data.nomLuthier||''}  ·  ${fmtDate(data.date)}`,px(infoX),py(tY+3*rH+rH*0.65));

    /* ─ légende poussé/tiré (petit cercle, haut à droite du titre) ─ */
    {
      const k=0.5522847498,lR_S=7,lX_S=W_S-MH_S-lR_S-2,lY_S=tY+tH/2;
      const lcx=px(lX_S),lcy=py(lY_S),lr=pr(lR_S);
      // fond blanc + demi-cercles colorés
      pdf.setFillColor(255,255,255);pdf.setDrawColor(...GRN2);pdf.setLineWidth(pr(1));
      pdf.circle(lcx,lcy,lr,'FD');
      pdf.setFillColor(...LGRE); // poussé lavande clair
      pdf.lines([[0,-k*lr,lr*(1-k),-lr,lr,-lr],[k*lr,0,lr,lr*(1-k),lr,lr]],lcx-lr,lcy,[1,1],'F',true);
      pdf.setFillColor(255,251,235); // tiré ambre très clair
      pdf.lines([[0,k*lr,lr*(1-k),lr,lr,lr],[k*lr,0,lr,-(lr*(1-k)),lr,-lr]],lcx-lr,lcy,[1,1],'F',true);
      pdf.setDrawColor(...GRN2);pdf.setLineWidth(pr(1.2));pdf.circle(lcx,lcy,lr,'S');
      pdf.setLineWidth(pr(0.4));pdf.line(lcx-lr,lcy,lcx+lr,lcy);
      // labels "Poussé / Tiré"
      pdf.setFontSize(Math.max(fs(6),3));pdf.setFont('helvetica','normal');pdf.setTextColor(...GRAY);
      pdf.text(t('svgPush').replace(/[↓↑]/g,'').trim(),px(lX_S-lR_S-2),py(lY_S-2),{align:'right'});
      pdf.text(t('svgPull').replace(/[↓↑]/g,'').trim(),px(lX_S-lR_S-2),py(lY_S+7),{align:'right'});
    }

    /* ─ "Haut (graves)" ─ */
    pdf.setFontSize(fs(8));pdf.setFont('helvetica','italic');pdf.setTextColor(...GRAY);
    pdf.text('Haut (graves)',px(rhX0-R_S-5),py(gridTop-HEADER_GAP_S+4));

    /* ─ dessin d'un bouton (cercle avec demi-cercles colorés poussé/tiré) ─ */
    function drawBtn(cx_s,cy_s,btn,btn2){
      const cx=px(cx_s),cy=py(cy_s),r=pr(R_S);
      const k=0.5522847498;
      // fond blanc de base
      pdf.setFillColor(255,255,255);pdf.setDrawColor(...CIRC_C);pdf.setLineWidth(pr(1));
      pdf.circle(cx,cy,r,'FD');
      // demi-cercle supérieur — poussé (lavande très clair)
      pdf.setFillColor(...LGRE);
      pdf.lines([[0,-k*r,r*(1-k),-r,r,-r],[k*r,0,r,r*(1-k),r,r]],cx-r,cy,[1,1],'F',true);
      // demi-cercle inférieur — tiré (ambre très clair)
      pdf.setFillColor(255,251,235);
      pdf.lines([[0,k*r,r*(1-k),r,r,r],[k*r,0,r,-(r*(1-k)),r,-r]],cx-r,cy,[1,1],'F',true);
      // contour du cercle (par-dessus les demi-cercles)
      pdf.setDrawColor(...GRN2);pdf.setLineWidth(pr(1.2));
      pdf.circle(cx,cy,r,'S');
      // ligne séparatrice poussé/tiré
      pdf.setLineWidth(pr(0.5));pdf.setDrawColor(...GRN2);
      pdf.line(cx-r,cy,cx+r,cy);
      // notes
      const has2=(btn2?(btn2.p||btn2.t):false);
      pdf.setFont('helvetica','bold');
      if(has2){
        const fB=Math.max(fs(6.5),2.5),fB2=Math.max(fs(5.5),2.5);
        if(btn.p){pdf.setFontSize(fB);pdf.setTextColor(...PUSH_C);pdf.text(displayNote(btn.p),cx,cy-pr(11),{align:'center'});}
        if(btn2?btn2.p:false){pdf.setFontSize(fB2);pdf.setTextColor(...GRN2);pdf.text(displayNote(btn2.p),cx,cy-pr(4),{align:'center'});}
        if(btn.t){pdf.setFontSize(fB);pdf.setTextColor(...PULL_C);pdf.text(displayNote(btn.t),cx,cy+pr(8),{align:'center'});}
        if(btn2?btn2.t:false){pdf.setFontSize(fB2);pdf.setTextColor(...PULL_C2);pdf.text(displayNote(btn2.t),cx,cy+pr(15),{align:'center'});}
      }else{
        const fB=Math.max(fs(8.5),2.5);
        if(btn.p){pdf.setFontSize(fB);pdf.setTextColor(...PUSH_C);pdf.text(displayNote(btn.p),cx,cy-pr(7),{align:'center'});}
        if(btn.t){pdf.setFontSize(fB);pdf.setTextColor(...PULL_C);pdf.text(displayNote(btn.t),cx,cy+pr(15),{align:'center'});}
      }
    }

    /* ─ colonnes MD (main droite) ─ */
    pdf.setFontSize(fs(8));pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
    pdf.text(t('svgRH'),px(rhX0+((nRH-1)*CGAP_S)/2),py(gridTop-HEADER_GAP_S+6),{align:'center'});
    rhR.forEach((row,ri)=>{
      const cx_s=rhX0+ri*CGAP_S,off=rhTS+(row.offset||0)*(BGAP_S/2);
      // bande de colonne alternée
      if((row.boutons||[]).length){
        const bY=gridTop+off+BGAP_S/2-R_S-3,bH=(row.boutons.length-1)*BGAP_S+2*(R_S+3);
        pdf.setFillColor(...(ri%2===0?[250,250,250]:[245,243,255]));
        pdf.rect(px(cx_s-R_S-3),py(bY),pr(2*(R_S+3)),pr(bH),'F');
      }
      // étiquette de colonne (pivotée 90°)
      pdf.setFontSize(Math.max(fs(6.5),3));pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
      pdf.text(row.nom||'',px(cx_s),py(gridTop-8),{angle:90,align:'left'});

      const sfx=ri===0?'':ri===1?"'":"''";
      (row.boutons||[]).forEach((btn,bi)=>{
        const cy_s=gridTop+off+bi*BGAP_S+BGAP_S/2;
        drawBtn(cx_s,cy_s,btn,null);
        // numéro de bouton (coin bas-gauche extérieur)
        pdf.setFontSize(Math.max(fs(5),2));pdf.setFont('helvetica','normal');pdf.setTextColor(...RULE_C);
        pdf.text((bi+1)+sfx,px(cx_s-R_S-1),py(cy_s+R_S+2),{align:'left'});
      });
    });

    /* ─ séparateur soufflet ─ */
    if(nLH>0){
      const sY1=gridTop-2,sY2=gridTop+maxB*BGAP_S+eOff+2;
      pdf.setDrawColor(...RULE_C);pdf.setLineWidth(pr(1.5));
      pdf.setLineDashPattern([pr(6),pr(4)],0);
      pdf.line(px(sepX),py(sY1),px(sepX),py(sY2));
      pdf.setLineDashPattern([],0);
      pdf.setFontSize(Math.max(fs(6),3));pdf.setFont('helvetica','bold');pdf.setTextColor(...GRAY);
      pdf.text(`— ${t('svgBellows')} —`,px(sepX),py((sY1+sY2)/2),{angle:90,align:'center'});
    }

    /* ─ colonnes MG (main gauche) ─ */
    if(nLH>0?maxLH>0:false){
      const rhBH=maxRH*BGAP_S+rhTS+rhBE,lhBH=maxLH*BGAP_S+lhTS+lhBE;
      const lhGT=gridTop+Math.round((rhBH-lhBH)/2);
      pdf.setFontSize(fs(8));pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
      pdf.text(t('svgLH'),px(lhX0+((nLH-1)*CGAP_S)/2),py(lhGT-HEADER_GAP_S+6),{align:'center'});
      lhSecs.forEach((sec,si)=>{
        const cx_s=lhX0+si*CGAP_S,off=lhTS+(sec.offset||0)*(BGAP_S/2);
        if(sec.items.length){
          const bY=lhGT+off+BGAP_S/2-R_S-3,bH=(sec.items.length-1)*BGAP_S+2*(R_S+3);
          pdf.setFillColor(...(si%2===0?[250,250,250]:[245,243,255]));
          pdf.rect(px(cx_s-R_S-3),py(bY),pr(2*(R_S+3)),pr(bH),'F');
        }
        pdf.setFontSize(Math.max(fs(6.5),3));pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
        const has2=(sec.items2?sec.items2.some(x=>x!=null):false);
        sec.items.forEach((btn,bi)=>{
          const cy_s=lhGT+off+bi*BGAP_S+BGAP_S/2;
          drawBtn(cx_s,cy_s,btn,has2?(sec.items2[bi]||null):null);
          if(sec.iTypes){
            const typ=sec.iTypes[bi],idx=sec.iIdxs?sec.iIdxs[bi]:'';
            pdf.setFontSize(Math.max(fs(5.5),2.5));pdf.setFont('helvetica','bold');
            pdf.setTextColor(typ==='B'?26:139,typ==='B'?76:105,typ==='B'?156:20);
            pdf.text(typ+String(idx),px(cx_s-R_S+4),py(cy_s-R_S-2));
          }
        });
      });
      // numéros à droite de la dernière colonne MG
      const lhLC=lhX0+(nLH-1)*CGAP_S;
      for(let i=0;i<maxLH;i++){
        pdf.setFontSize(Math.max(fs(5.5),2));pdf.setFont('helvetica','normal');pdf.setTextColor(...RULE_C);
        pdf.text(String(i+1),px(lhLC+R_S+5),py(lhGT+i*BGAP_S+BGAP_S/2+3));
      }
    }

    /* ─ "↑ Aigus" en bas ─ */
    pdf.setFontSize(fs(8));pdf.setFont('helvetica','italic');pdf.setTextColor(...GRAY);
    pdf.text('↑ Aigus',px(W_S/2),py(H_S-MB_S+12),{align:'center'});

    yPos+=planH+5;
  }

  // ── AutoTable defaults ─────────────────────────────
  function tableDefaults(extra){
    return Object.assign({
      margin:{left:ML,right:MR},theme:'grid',
      rowPageBreak:'avoid',
      styles:{fontSize:7,cellPadding:[1.8,3],textColor:[30,30,30],
              lineColor:RULE_C,lineWidth:0.22},
      headStyles:{fillColor:GRN,textColor:[255,255,255],fontStyle:'bold',fontSize:7},
      alternateRowStyles:{fillColor:ALT},
    },extra);
  }
  // Returns MIDI note as string cell, or '—'

  // ── Sommier as vector table (sharp text, copy-pastable) ───
  const PULL_BG=[255,252,235];

    secLabel(title,yPos+4);yPos+=6;
    sections.forEach(function(row){
      const btns=row.items||row.boutons||[];
      if(!btns.length)return;
      if(yPos+30>CB){pdf.addPage();yPos=MT;}
      const n=btns.length;
      const labelW=18;
      const numRow=[{content:'#',styles:{fillColor:LGRE,textColor:GRAY,fontStyle:'bold',halign:'center',cellWidth:labelW}}]
        .concat(btns.map(function(_,i){return{content:String(i+1),styles:{halign:'center',textColor:GRAY,fontSize:6}};}));
      const pushLabel=t('svgPush').replace('\u2193','v').replace('\u2191','^');
      const pullLabel=t('svgPull').replace('\u2191','^').replace('\u2193','v');
      const pushRow=[{content:pushLabel,styles:{fillColor:LGRE,textColor:PUSH_C,fontStyle:'bold',halign:'center',cellWidth:labelW}}]
        .concat(btns.map(function(b){return{content:(b?b.p:false)?displayNote(b.p):'-',styles:{halign:'center',fontStyle:'bold',textColor:PUSH_C}};}));
      const pullRow=[{content:pullLabel,styles:{fillColor:PULL_BG,textColor:PULL_C,fontStyle:'bold',halign:'center',cellWidth:labelW}}]
        .concat(btns.map(function(b){return{content:(b?b.t:false)?displayNote(b.t):'-',styles:{halign:'center',fontStyle:'bold',textColor:PULL_C}};}));
      pdf.autoTable(tableDefaults({
        startY:yPos,pageBreak:'avoid',
        head:[[{content:row.nom||'',colSpan:n+1,styles:{fillColor:GRN2,textColor:[255,255,255],fontSize:7.5,fontStyle:'bold',halign:'center'}}],numRow],
        body:[pushRow,pullRow],
        columnStyles:{0:{cellWidth:labelW,fillColor:LGRE}},
      }));
      yPos=pdf.lastAutoTable.finalY+3;
    });
    yPos+=2;
  }

  // ══════════════════════════════════════════════════
  // PAGE 1 — Cover header
  // ══════════════════════════════════════════════════
  hRule(yPos,GRN,0.6);yPos+=5;

  pdf.setFontSize(7);pdf.setFont('helvetica','normal');pdf.setTextColor(...GRAY);
  const badge=which==='visual'?t('pdfClientTitle'):t('pdfMfrTitle');
  pdf.text(badge.toUpperCase(),PW-MR,yPos+8,{align:'right'});
  yPos+=12;

  // Metadata row
  const meta=[
    `${t('svgLuthier').toUpperCase()}  ${data.nomLuthier||'—'}`,

    fmtDate(data.date),
  ].join('    ·    ');
  pdf.setFontSize(7);pdf.setTextColor(...GRAY);
  pdf.text(meta,ML,yPos);yPos+=5;
  hRule(yPos,GRN,0.4);yPos+=8;

  // ── Plan de clavier — dessiné nativement avec les primitives jsPDF (texte sélectionnable) ──
  drawPlanNative(data);

  // ══════════════════════════════════════════════════
  // FABRICANT PAGES
  // ══════════════════════════════════════════════════
  if(which==='fabricant'){

    // ── Page: Bon de commande ──────────────────────
    pdf.addPage();yPos=MT;
    const os=data.orderSpecs||{},date=fmtDate(data.date);
    const totalBtns=data.droite.reduce((s,r)=>s+r.boutons.length,0);
    const totalAnches=totalBtns*(data.nbVoix||2)
      +(data.basses?data.basses.length*2:0)*(data.nbVoixBasses||2)
      +(data.accords?data.accords.length*2:0)*(data.nbVoixAccords||2);
    const midi=data.showMidi;

    pdf.setFontSize(9);pdf.setFont('helvetica','bold');pdf.setTextColor(...GRN);
    pdf.text(t('pdfMfrTitle').toUpperCase(),ML,yPos+5);
    pdf.setFontSize(7);pdf.setFont('helvetica','normal');pdf.setTextColor(...GRAY);

    pdf.text(fmtDate(data.date),PW-MR,PH-6.5,{align:'right'});
  }

  const fileName=`${(data.nomInstrument||'accordeon').replace(/[^a-zA-Z0-9_\-]/g,'_')}_${which==='visual'?'plan':'fabricant'}.pdf`;
  pdf.save(fileName);
}

function convertFlatsToSharps(){
  if(!confirm(t('flatToSharpConfirm')))return;
  // Flat→Sharp maps (French and English roots)
  const F2S={
    'Réb':'Do#','Mib':'Ré#','Solb':'Fa#','Lab':'Sol#','Sib':'La#','Dob':'Si',
    'Db':'C#','Eb':'D#','Gb':'F#','Ab':'G#','Bb':'A#','Cb':'B'
  };
  function convertNote(s){
    if(!s)return s;
    // Slash-separated notes: convert each part recursively (handles octave in each part)
    if(s.includes('/')){return s.split('/').map(p=>convertNote(p.trim())).join('/');}
    // Chord name: "Réb3 Maj" → "Do#3 Maj"
    const sp=s.indexOf(' ');
    if(sp>0){const root=s.slice(0,sp),qual=s.slice(sp);return convertNote(root)+qual;}
    // Simple note possibly with octave suffix: "Réb3" → "Do#3"
    const m=s.match(/^(.+?)(\d)$/);
    if(m){const conv=F2S[m[1]];if(conv!==undefined)return conv+m[2];}
    return F2S[s]||s;
  }
  function convertRow(arr){return (arr||[]).map(b=>({p:convertNote(b.p),t:convertNote(b.t)}));}
  const n=CFG.notes;
  n.droite=(n.droite||[]).map(convertRow);
  ['basses','basses2','basses3','accords','accords2','accords3','terzetto'].forEach(k=>{n[k]=convertRow(n[k]);});
  generateInputGrid();onAnyChange();forcePreview();
}

function _doTranspose(semi){
  function shiftNoteStr(s){
    if(!s)return s;
    if(s.includes('/')){return s.split('/').map(p=>shiftNoteStr(p.trim())).join('/');}
    const sp=s.indexOf(' ');
    if(sp>0){const root=s.slice(0,sp),qual=s.slice(sp);return shiftNoteStr(root)+qual;}
    return shiftNote(s,semi)||s;
  }
  function shiftRow(arr){return (arr||[]).map(b=>({p:shiftNoteStr(b.p),t:shiftNoteStr(b.t)}));}
  const n=CFG.notes;
  n.droite=(n.droite||[]).map(shiftRow);
  ['basses','basses2','basses3','accords','accords2','accords3','terzetto'].forEach(k=>{n[k]=shiftRow(n[k]);});
  generateInputGrid();onAnyChange();forcePreview();
}

function renderKbdCompare(){
  const idA=document.getElementById('kc-sel-a').value;
  const idB=document.getElementById('kc-sel-b').value;
  if(!idA||!idB){alert(t('compareKbdPickBoth'));return;}
  const recA=DB.get(idA),recB=DB.get(idB);
  if(!recA||!recB)return;
  // Convert DB record to the format expected by buildSVG_Plan
  function toData(rec){
    const nRec=rec.notes||{};
    const droiteNotes=nRec.droite||[];
    const droite=(rec.rangees||[]).map((row,ri)=>({
      nom:row.nom||`Rangée ${ri+1}`,
      register:row.register||'N',
      offset:row.offset||0,
      boutons:droiteNotes[ri]||[]
    }));
    return{
      nomInstrument:rec.nom||'',
      nomJoueur:rec.joueur||'',
      nomLuthier:rec.luthier||"Ewen d'Aviau",
      date:rec.dateCreation||'',
      nbVoix:rec.nbVoix||2,
      voixPerRow:rec.voixPerRow||[],
      nbVoixAccords:rec.nbVoixAccords||2,
      nbVoixBasses:rec.nbVoixBasses||2,
      lhDisplayMode:rec.lhDisplayMode||'separated',
      lhPairedDir:rec.lhPairedDir||'vertical',
      lhRows:rec.lhRows||[{nb:6,offset:0}],
      lhPairsPerRow:rec.lhPairsPerRow||2,
      lhBassFirst:rec.lhBassFirst||false,
      showMidi:false,
      droite,
      basses:nRec.basses||[],
      basses2:nRec.basses2||[],
      basses3:nRec.basses3||[],
      accords:nRec.accords||[],
      accords2:nRec.accords2||[],
      accords3:nRec.accords3||[],
      terzetto:nRec.terzetto||[],
      orderSpecs:rec.orderSpecs||{},
    };
  }
  // Render both SVG plans
  try{document.getElementById('kc-pane-a').innerHTML=buildSVG_Plan(toData(recA),false);}
  catch(e){document.getElementById('kc-pane-a').innerHTML=`<p style="color:red;padding:8px">${escHTML(e.message)}</p>`;}
  try{document.getElementById('kc-pane-b').innerHTML=buildSVG_Plan(toData(recB),false);}
  catch(e){document.getElementById('kc-pane-b').innerHTML=`<p style="color:red;padding:8px">${escHTML(e.message)}</p>`;}
  // ── Enharmonic note comparison ────────────────────────────────
  const _CHROMA={
    'Do':0,'Do#':1,'Réb':1,'Ré':2,'Ré#':3,'Mib':3,'Mi':4,
    'Fa':5,'Fa#':6,'Solb':6,'Sol':7,'Sol#':8,'Lab':8,'La':9,'La#':10,'Sib':10,'Si':11,'Dob':11,
    'C':0,'C#':1,'Db':1,'D':2,'D#':3,'Eb':3,'E':4,'Fb':4,
    'F':5,'F#':6,'Gb':6,'G':7,'G#':8,'Ab':8,'A':9,'A#':10,'Bb':10,'B':11,'Cb':11
  };
  function noteCmp(a,b){
    a=(a||'').trim();b=(b||'').trim();
    if(a===b)return 'same';
    if(!a?!b:false)return 'same';
    if(!a||!b)return 'diff';
    // Slash-separated notes (e.g. "Do#/Mi/Sol")
    if(a.includes('/')||b.includes('/')){
      const pa=a.split('/').map(x=>x.trim()),pb=b.split('/').map(x=>x.trim());
      if(pa.length!==pb.length)return 'diff';
      let hasEnh=false;
      for(let i=0;i<pa.length;i++){const r=noteCmp(pa[i],pb[i]);if(r==='diff')return 'diff';if(r==='enh')hasEnh=true;}
      return hasEnh?'enh':'same';
    }
    // Extract root (strip trailing octave digit) and chord quality
    function parse(s){const sp=s.indexOf(' ');if(sp>0)return{root:s.slice(0,sp).replace(/\d+$/,''),qual:s.slice(sp+1)};return{root:s.replace(/\d+$/,''),qual:''};}
    const pA=parse(a),pB=parse(b);
    const cA=_CHROMA[pA.root],cB=_CHROMA[pB.root];
    if(cA===undefined||cB===undefined)return 'diff';
    if(pA.qual!==pB.qual)return 'diff';
    if(cA!==cB)return 'diff';
    return pA.root===pB.root?'same':'enh';
  }
  // Build diff summary as {cls, text} entries
  const diffs=[];
  function addCmp(label,vA,vB){
    const r=noteCmp(vA,vB);
    if(r==='same')return;
    const a=escHTML(vA||'—'),b=escHTML(vB||'—');
    if(r==='enh')diffs.push({cls:'diff-enh',text:`${label}: ${a} ≈ ${b} (${t('compareKbdEnh')})`});
    else diffs.push({cls:'diff-diff',text:`${label}: ${a} ≠ ${b}`});
  }
  const nA=recA.notes||{},nB=recB.notes||{};
  // RH droite
  const droiteA=nA.droite||[],droiteB=nB.droite||[];
  const nRowA=droiteA.length,nRowB=droiteB.length;
  if(nRowA!==nRowB)diffs.push({cls:'diff-diff',text:`${t('compareKbdRH')}: ${nRowA} rangées ≠ ${nRowB} rangées`});
  for(let ri=0;ri<Math.min(nRowA,nRowB);ri++){
    const rA=droiteA[ri]||[],rB=droiteB[ri]||[];
    const nBtnA=rA.length,nBtnB=rB.length;
    if(nBtnA!==nBtnB)diffs.push({cls:'diff-diff',text:`${t('compareKbdRH')}${ri+1}: ${nBtnA} boutons ≠ ${nBtnB} boutons`});
    for(let bi=0;bi<Math.min(nBtnA,nBtnB);bi++){
      const bA=rA[bi]||{},bB=rB[bi]||{};
      addCmp(`R${ri+1}·${bi+1} ${t('compareKbdPush')}`,bA.p,bB.p);
      addCmp(`R${ri+1}·${bi+1} ${t('compareKbdPull')}`,bA.t,bB.t);
    }
  }
  // LH basses
  const bassA=nA.basses||[],bassB=nB.basses||[];
  if(bassA.length!==bassB.length)diffs.push({cls:'diff-diff',text:`${t('compareKbdLH')} basses: ${bassA.length} ≠ ${bassB.length}`});
  for(let i=0;i<Math.min(bassA.length,bassB.length);i++){
    const bA=bassA[i]||{},bB=bassB[i]||{};
    addCmp(`${t('compareKbdLH')} B${i+1} ${t('compareKbdPush')}`,bA.p,bB.p);
    addCmp(`${t('compareKbdLH')} B${i+1} ${t('compareKbdPull')}`,bA.t,bB.t);
  }
  // LH accords
  const accA=nA.accords||[],accB=nB.accords||[];
  if(accA.length!==accB.length)diffs.push({cls:'diff-diff',text:`${t('compareKbdLH')} accords: ${accA.length} ≠ ${accB.length}`});
  for(let i=0;i<Math.min(accA.length,accB.length);i++){
    const bA=accA[i]||{},bB=accB[i]||{};
    addCmp(`${t('compareKbdLH')} A${i+1} ${t('compareKbdPush')}`,bA.p,bB.p);
    addCmp(`${t('compareKbdLH')} A${i+1} ${t('compareKbdPull')}`,bA.t,bB.t);
  }
  const diffEl=document.getElementById('kc-diff');
  if(diffs.length===0){
    diffEl.innerHTML=`<span class="diff-same">✓ ${t('compareKbdSame')}</span>`;
  } else {
    diffEl.innerHTML=`<div class="diff-title">⚠ ${t('compareKbdDiff')}</div>`+
      diffs.map(d=>`<div class="${d.cls}">• ${d.text}</div>`).join('');
  }
  diffEl.style.display='block';
}

function loadFromFile(evt){
  const file=(evt.target.files?evt.target.files[0]:false);
  if(!file)return;
  evt.target.value=''; // reset so same file can be re-loaded
  const reader=new FileReader();
  reader.onload=e=>{
    try{
      let data=JSON.parse(e.target.result);
      // Accept both a single object and an array
      if(!Array.isArray(data))data=[data];
      data.forEach((rec,i)=>{if(!rec.id)rec.id=Date.now().toString(36)+'_'+i+'_'+Math.random().toString(36).slice(2);});
      const existing=DB.load();
      // Merge: skip duplicates by id, add new ones at top
      const ids=new Set(existing.map(r=>r.id));
      const fresh=data.filter(r=>!ids.has(r.id));
      DB.save([...fresh,...existing]);
      renderLibrary();
      if(data.length){loadFromDB(data[0]);scheduleLivePreview();}
      alert(t('loadedFrom')(data.length));
    }catch(err){alert(t('importError')+err.message);}
  };
  reader.readAsText(file);
}
function triggerDownload(content,filename,mime){
  const blob=new Blob([content],{type:mime});
  const url=URL.createObjectURL(blob);
  const a=document.createElement('a');a.href=url;a.download=filename;
  document.body.appendChild(a);a.click();
  setTimeout(()=>{document.body.removeChild(a);URL.revokeObjectURL(url);},100);
}
