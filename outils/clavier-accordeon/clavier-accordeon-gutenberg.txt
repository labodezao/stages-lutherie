  /* ═══════════════════════════════════════════════════
     CUSTOM PRESETS — lus depuis <script id="cw-custom-presets">
     Fusionnés avec PRESET_CATALOG + PRESETS_DATA au chargement.

  function readCustomPresets(){
    try {
      var tag = document.getElementById('cw-custom-presets');
      if(!tag){ return []; }
      return JSON.parse(tag.textContent) || [];
    } catch(e){
      console.warn('cw-custom-presets: JSON invalide —', e.message);
      return [];
    }
  }

  function mergeCustomPresets(){
    var customs = readCustomPresets();
    customs.forEach(function(c){
      if(!c.id){ return; }
      if(!c.data){ return; }
      var exists = PRESET_CATALOG.some(function(p){ return p.id === c.id; });
      if(!exists){
        PRESET_CATALOG.push({
          id: c.id,
          label: c.label || c.id,
          category: c.category || 'Personnalisés'
        });
      }
      PRESETS_DATA[c.id] = c.data;
    });
  }

  function exportAsPreset(){
    var snap = cwSnapshotPreset();
    if(!snap){ showToast('Aucun modèle chargé.'); return; }
    var rawId = ((snap.nom || snap.id || 'preset').replace(/\s+/g,'_').replace(/[^A-Za-z0-9_\-]/g,'').substring(0,40)) || ('preset_' + Date.now());
    var entry = [{
      id: rawId,
      label: snap.nom || 'Préset personnalisé',
      category: 'Personnalisés',
      data: snap
    }];
    var json = JSON.stringify(entry, null, 2);
    var area = $('cw-preset-export-area');
    if(area){
      area.value = json;
      area.style.display = 'block';
      area.focus();
      area.select();
    }
    if(navigator.clipboard){
      navigator.clipboard.writeText(json).then(function(){
        showToast('✔ JSON copié ! Collez-le dans le bloc cw-custom-presets de la page WordPress.');
      }).catch(function(){
        showToast('✔ JSON prêt — sélectionnez et copiez le texte ci-dessous.');
      });
    } else {
      showToast('✔ JSON prêt — sélectionnez et copiez le texte ci-dessous.');
    }
  }

  function initNotePicker(){

  /* Parse "Sol4" → {note:'Sol', oct:'4'} */
  function parseNoteOct(v){
    if(!v) return {note:'', oct:''};
    var m = v.trim().match(/^(.+?)([2-6])$/);
    return m ? {note:m[1], oct:m[2]} : {note:v.trim(), oct:''};
  }

  function openNotePicker(editId, evt, initialSlot){
    if(!currentPreset || !editId) return;
    var parts = editId.split(':');
    var btn = {p:'', t:''}, title = '';
    if(parts[0] === 'rh'){
      var ri = parseInt(parts[1]), bi = parseInt(parts[2]);
      var row = (currentPreset.rangees || [])[ri];
      var rowNotes = ((currentPreset.notes || {}).droite || [])[ri] || [];
      btn = rowNotes[bi] || {p:'', t:''};
      title = (row ? row.nom : 'Rangée '+(ri+1)) + ' — Bouton '+(bi+1);
    } else {
      var src = parts[0], idx = parseInt(parts[1]);
      var arr = ((currentPreset.notes || {})[src] || []);
      btn = arr[idx] || {p:'', t:''};
      var lbl = src === 'basses' ? 'Basse' : src === 'accords' ? 'Accord' : 'Terzetto';
      title = lbl + ' '+(idx+1);
    }
    var pParsed = parseNoteOct(btn.p || '');
    var tParsed = parseNoteOct(btn.t || '');
    _cwEdit = {
      editId: editId,
      slot: 'push',
      pushVal: btn.p || '', pullVal: btn.t || '',
      pushNote: pParsed.note, pushOct: pParsed.oct,
      pullNote: tParsed.note, pullOct: tParsed.oct
    };
    $('cw-np-btn-title').textContent = title;
    cwSwitchSlot(initialSlot || 'push');
    cwPickerHL();
    cwPickerUpdateBadges();

    var pk = $('cw-note-picker');
    pk.style.display = 'block';
    /* Position popup near click but keep in viewport */
    var pw = pk.offsetWidth || 320, ph = pk.offsetHeight || 340;
    var x = evt.clientX + 12, y = evt.clientY + 12;
    if(x + pw > window.innerWidth  - 8) x = evt.clientX - pw - 8;
    if(y + ph > window.innerHeight - 8) y = evt.clientY - ph - 8;
    pk.style.left = Math.max(8, x) + 'px';
    pk.style.top  = Math.max(8, y) + 'px';
  }

  function closeNotePicker(){
    var pk = $('cw-note-picker');
    pk.style.display = 'none';
    _cwEdit = null; _cwNote = ''; _cwOct = '';
  }

  function cwSwitchSlot(slot){
    if(!_cwEdit) return;
    _cwEdit.slot = slot;
    if(slot === 'push'){
      _cwNote = _cwEdit.pushNote; _cwOct = _cwEdit.pushOct;
    } else {
      _cwNote = _cwEdit.pullNote; _cwOct = _cwEdit.pullOct;
    }
    widget.querySelectorAll('.cw-np-slot-btn').forEach(function(b, i){
      b.classList.toggle('cw-np-slot-active', (i === 0 ? 'push' : 'pull') === slot);
    });
    cwPickerHL();
    cwPickerUpdatePreview();
  }

  function cwPickerSetNote(n){
    _cwNote = n;
    if(_cwEdit){ if(_cwEdit.slot === 'push') _cwEdit.pushNote = n; else _cwEdit.pullNote = n; }
    cwPickerHL(); cwPickerUpdatePreview();
    if(_cwNote && _cwOct) cwPickerAutoApplySlot();
  }

  function cwPickerSetOctave(o){
    _cwOct = o;
    if(_cwEdit){ if(_cwEdit.slot === 'push') _cwEdit.pushOct = o; else _cwEdit.pullOct = o; }
    cwPickerHL(); cwPickerUpdatePreview();
    if(_cwNote && _cwOct) cwPickerAutoApplySlot();
  }

  /* Auto-apply current slot and switch to the other slot if it is empty */
  function cwPickerAutoApplySlot(){
    if(!_cwEdit) return;
    var val = _cwNote + _cwOct;
    if(_cwEdit.slot === 'push'){
      _cwEdit.pushVal = val;
      _cwEdit.pushNote = _cwNote; _cwEdit.pushOct = _cwOct;
      cwPickerUpdateBadges();
      /* Switch to pull if pull is still empty */
      if(!_cwEdit.pullVal){ cwSwitchSlot('pull'); return; }
    } else {
      _cwEdit.pullVal = val;
      _cwEdit.pullNote = _cwNote; _cwEdit.pullOct = _cwOct;
      cwPickerUpdateBadges();
      /* Switch to push if push is still empty */
      if(!_cwEdit.pushVal){ cwSwitchSlot('push'); return; }
    }
    /* Both slots have a value — apply and close */
    cwPickerApply();
  }

  function cwPickerClear(){
    if(!_cwEdit) return;
    _cwNote = ''; _cwOct = '';
    if(_cwEdit.slot === 'push'){
      _cwEdit.pushNote = ''; _cwEdit.pushOct = ''; _cwEdit.pushVal = '';
    } else {
      _cwEdit.pullNote = ''; _cwEdit.pullOct = ''; _cwEdit.pullVal = '';
    }
    cwPickerHL(); cwPickerUpdatePreview(); cwPickerUpdateBadges();
  }

  function cwPickerApply(){
    if(!_cwEdit || !currentPreset) return;
    /* Use note+oct only when both components are present, otherwise fall back to stored value */
    var p = (_cwEdit.pushNote && _cwEdit.pushOct) ? (_cwEdit.pushNote + _cwEdit.pushOct) : (_cwEdit.pushVal || '');
    var t = (_cwEdit.pullNote && _cwEdit.pullOct) ? (_cwEdit.pullNote + _cwEdit.pullOct) : (_cwEdit.pullVal || '');
    var notes = currentPreset.notes;
    if(!notes) { notes = {}; currentPreset.notes = notes; }
    var eid = _cwEdit.editId;
    var parts2 = eid.split(':');
    if(parts2[0] === 'rh'){
      var ri2 = parseInt(parts2[1]), bi2 = parseInt(parts2[2]);
      if(!notes.droite)          notes.droite          = [];
      if(!notes.droite[ri2])     notes.droite[ri2]     = [];
      if(!notes.droite[ri2][bi2])notes.droite[ri2][bi2]= {};
      notes.droite[ri2][bi2].p = p;
      notes.droite[ri2][bi2].t = t;
    } else {
      var src3 = parts2[0], idx3 = parseInt(parts2[1]);
      if(!notes[src3])       notes[src3]       = [];
      if(!notes[src3][idx3]) notes[src3][idx3] = {};
      notes[src3][idx3].p = p;
      notes[src3][idx3].t = t;
    }
    closeNotePicker();
    refreshSVG();
    renderRHTables();
    renderLHTables();
  }

  function cwPickerHL(){
    widget.querySelectorAll('#cw-np-naturals .cw-np-btn, #cw-np-sharps .cw-np-btn, #cw-np-flats .cw-np-btn').forEach(function(b){
      b.classList.toggle('cw-sel', b.textContent === _cwNote);
    });
    widget.querySelectorAll('#cw-np-octaves .cw-np-oct-btn').forEach(function(b){
      b.classList.toggle('cw-sel', b.textContent === _cwOct);
    });
  }

  function cwPickerUpdatePreview(){
    var txt = (_cwNote + _cwOct) || '— sélectionnez note + octave —';
    $('cw-np-preview-text').textContent = txt;
  }

  function cwPickerUpdateBadges(){
    if(!_cwEdit) return;
    $('cw-np-badge-push').textContent = _cwEdit.pushVal || '';
    $('cw-np-badge-pull').textContent = _cwEdit.pullVal || '';
  }

  function escSVG(s){

    return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');

  function svgBtnVal(v, cx, baseY, color){
    if(!v) return '';
    var disp = displayNote(v);
    if(!disp.includes('/')){
      var fs = disp.length > 8 ? 5.5 : disp.length > 5 ? 6.8 : 8;
      return tx(cx, baseY, escSVG(disp), fs, color, 'bold', 'middle');
    }
    var parts = disp.split('/').map(function(p){ return p.trim(); }).filter(Boolean);
    var lineH = 6.5, fs2 = 5.5;
    var startY = baseY - ((parts.length - 1) * lineH) / 2;
    var out = '<text font-size="'+fs2+'" fill="'+color+'" font-weight="bold" text-anchor="middle">';
    parts.forEach(function(p, i){ out += '<tspan x="'+cx+'" y="'+(startY + i * lineH)+'">'+escSVG(p)+'</tspan>'; });
    return out + '</text>';

        var cx = lhX0 + si * CGAP;
        var off = lhTopShift + (sec.offset||0) * (BGAP/2);
        var has2 = sec.items2 ? sec.items2.some(function(x){ return x != null; }) : false;
        if(sec.items.length){
          var cy0 = lhGridTop + off + BGAP/2;
          var bandY = cy0 - R - 3;
          var bandH = (sec.items.length-1)*BGAP + 2*(R+3);
          s += '<rect x="'+(cx-R-3)+'" y="'+bandY+'" width="'+((R+3)*2)+'" height="'+bandH+'" rx="3" fill="'+(si%2===0?band1:band2)+'" opacity=".6"/>';
        }
        sec.items.forEach(function(btn, bi){
          var cy = lhGridTop + off + bi*BGAP + BGAP/2;
          var lhEditId = null;
          if(sec.interleavedTypes){
            var eTyp = sec.interleavedTypes[bi] === 'B' ? 'basses' : 'accords';
            var eIdx = sec.interleavedIndices ? (sec.interleavedIndices[bi] - 1) : bi;
            lhEditId = eTyp + ':' + eIdx;
          } else if(sec.src){
            lhEditId = sec.src + ':' + (sec.startIdx + bi);
          }
          s += drawBtn(cx, cy, btn, has2 ? (sec.items2[bi]||null) : null, lhEditId);
          if(sec.interleavedTypes){
            var typ = sec.interleavedTypes[bi];
            var idx = sec.interleavedIndices ? sec.interleavedIndices[bi] : '';
            s += tx(cx-R+4, cy-R-2, typ+idx, 5.5, typ==='B'?bassLbl:accLbl, 'bold', 'start');
          }
        });
      });
      var lhLastCx = lhX0 + (nLH-1)*CGAP;
      for(var i4 = 0; i4 < maxLHBtns; i4++)
        s += tx(lhLastCx+R+5, lhGridTop+i4*BGAP+BGAP/2+3, i4+1, 6, numTx, 'normal', 'start');
    }}

    s += '<text x="'+(W/2)+'" y="'+(H-MB+12)+'" font-size="8.5" fill="'+textColor+'" font-style="italic" text-anchor="middle">\u2191 Aigus</text>';
    s += '<line x1="'+MH+'" y1="'+(H-MB+20)+'" x2="'+(W-MH)+'" y2="'+(H-MB+20)+'" stroke="'+ruleColor+'" stroke-width="1"/>';
    s += tx(MH, H-MB+28, escSVG(data.nomLuthier)+' \u2014 ewendaviau.com', 7, subtl, 'normal', 'start');

    s += svg_close();
    return s;
  }

  /* ═══════════════════════════════════════════════════
     toData — convert preset JSON to rendering format
     ═══════════════════════════════════════════════════ */
  function toData(rec){
    var nRec = rec.notes || {};
    var droiteNotes = nRec.droite || [];
    var droite = (rec.rangees || []).map(function(row, ri){
      return {
        nom: row.nom || ('Rangée '+(ri+1)),
        register: row.register || 'N',
        offset: row.offset || 0,
        boutons: droiteNotes[ri] || []
      };
    });
    return {
      nomInstrument: rec.nom || rec.label || '',
      nomJoueur: rec.joueur || '',
      nomLuthier: rec.luthier || "Ewen d'Aviau",
      date: rec.dateCreation || '',
      nbVoix: rec.nbVoix || 2,
      voixPerRow: rec.voixPerRow || [],
      nbVoixAccords: rec.nbVoixAccords || 2,
      nbVoixBasses: rec.nbVoixBasses || 2,
      lhDisplayMode: rec.lhDisplayMode || 'separated',
      lhPairedDir: rec.lhPairedDir || 'vertical',
      lhRows: rec.lhRows || [{nb: (nRec.basses||[]).length || 1, offset: 0}],
      lhPairsPerRow: rec.lhPairsPerRow || 2,
      lhBassFirst: rec.lhBassFirst || false,
      showMidi: false,
      droite: droite,
      basses: nRec.basses || [],
      basses2: nRec.basses2 || [],
      basses3: nRec.basses3 || [],
      accords: nRec.accords || [],
      accords2: nRec.accords2 || [],
      accords3: nRec.accords3 || [],
      terzetto: nRec.terzetto || [],
      orderSpecs: rec.orderSpecs || {}
    };
  }

  function loadPreset(id){
    if(!id){ currentPreset = null; currentData = null; renderAll(); return; }
    var raw = PRESETS_DATA[id];
    if(!raw){ $('cw-svg-wrap').innerHTML = '<p style="color:#c0392b;padding:12px;">Preset introuvable : ' + id + '</p>'; return; }
    var json = JSON.parse(JSON.stringify(raw)); // deep clone
    var cat = PRESET_CATALOG.find(function(p){ return p.id === id; });
    if(cat) json.label = cat.label;
    currentPreset = json;
    applyUIToPreset();
    renderAll();
  }

  function onRowCfgChange(e){
    var inp = e.target;
    var ri = parseInt(inp.getAttribute('data-row'));
    var field = inp.getAttribute('data-field');
    if(!currentPreset || !currentPreset.rangees[ri]) return;
    if(field === 'nom'){
      currentPreset.rangees[ri].nom = inp.value;
    } else if(field === 'nb'){
      var newNb = Math.max(1, parseInt(inp.value)||1);
      var oldNb = currentPreset.rangees[ri].nb;
      currentPreset.rangees[ri].nb = newNb;
      // Adjust notes array
      var notes = currentPreset.notes.droite[ri] || [];
      if(newNb > oldNb){
        for(var k = oldNb; k < newNb; k++) notes.push({p:'',t:''});
      } else {
        notes.length = newNb;
      }
      currentPreset.notes.droite[ri]=notes;
    } else if(field==='offset'){
      currentPreset.rangees[ri].offset=Math.max(-30,Math.min(30,parseInt(inp.value)||0));
    }
    renderAll();
  }

  function onNoteChange(e){
    var inp = e.target;
    var section = inp.getAttribute('data-section');
    var bi = parseInt(inp.getAttribute('data-btn'));
    var dir = inp.getAttribute('data-dir');
    var val = inp.value.trim();
    if(!currentPreset) return;
    var notes = currentPreset.notes;
    if(section === 'rh'){
      var ri = parseInt(inp.getAttribute('data-row'));
      if(!notes.droite[ri]) notes.droite[ri] = [];
      if(!notes.droite[ri][bi]) notes.droite[ri][bi] = {p:'',t:''};
      notes.droite[ri][bi][dir] = val;
    } else if(section === 'basses'){
      if(!notes.basses[bi]) notes.basses[bi] = {p:'',t:''};
      notes.basses[bi][dir] = val;
    } else if(section === 'accords'){
      if(!notes.accords[bi]) notes.accords[bi] = {p:'',t:''};
      notes.accords[bi][dir] = val;
    }
    // Re-render SVG only (not the tables, to keep focus)
    applyUIToPreset();
    currentData = toData(currentPreset);
    currentData.nomJoueur = $('cw-nom-joueur').value.trim();
    var nomI = $('cw-nom-instrument').value.trim();
    if(nomI) currentData.nomInstrument = nomI;
    try {
      $('cw-svg-wrap').innerHTML = buildSVG_Plan(currentData, darkMode);
      bindSVGEditing();
    } catch(err){
      $('cw-svg-wrap').innerHTML = '<p style="color:#c0392b;padding:12px;">Erreur de rendu : '+err.message+'</p>';
    }
  }

  function buildExportText(){
    if(!currentPreset) return '';
    applyUIToPreset();
    var pr=currentPreset;
    var nom=($('cw-nom-instrument').value.trim()||pr.nom||pr.label||'Sans titre');
    var joueur=$('cw-nom-joueur').value.trim();
    var date=new Date().toLocaleDateString('fr-FR');
    var pad=function(s,n){s=String(s);while(s.length<n)s=' '+s;return s;};
    var lines=['══════════════════════════════════','  '+nom];
    if(joueur)lines.push('  Joueur : '+joueur);
    lines.push('  '+date,'══════════════════════════════════','','MAIN DROITE','──────────────────────────────────');
    (pr.rangees||[]).forEach(function(row,ri){
      var label=row.nom||('Rangée '+(ri+1));
      if(row.offset) label+=' (décalage '+(row.offset>0?'+':'')+row.offset+')';
      lines.push('  '+label);
      var btns=((pr.notes||{}).droite||[])[ri]||[];
      for(var bi=0;bi<(row.nb||btns.length);bi++){
        var b=btns[bi]||{p:'',t:''};
        lines.push('    '+pad(bi+1,2)+'.  ↓ '+pad(b.p||'–',8)+'  ↑ '+(b.t||'–'));
      }
    });
    var basses=(pr.notes||{}).basses||[];
    var accords=(pr.notes||{}).accords||[];
    if(basses.length||accords.length){
      lines.push('','MAIN GAUCHE','──────────────────────────────────');
      if(basses.length){
        var _lhR0=pr.lhRows?pr.lhRows[0]:null,_lhO=_lhR0?(_lhR0.offset||0):0;
        var lhOff=_lhO?(' (décalage '+(_lhO>0?'+':'')+_lhO+')'):'';
        lines.push('  Basses'+lhOff);
        basses.forEach(function(b,i){lines.push('    '+pad(i+1,2)+'.  ↓ '+pad(b.p||'–',8)+'  ↑ '+(b.t||'–'));});
      }
      if(accords.length){
        lines.push('  Accords');
        accords.forEach(function(a,i){lines.push('    '+pad(i+1,2)+'.  ↓ '+pad(a.p||'–',8)+'  ↑ '+(a.t||'–'));});
      }
    }
    lines.push('','══════════════════════════════════');
    return lines.join('\n');
  }

  function exportText(){
    var txt=buildExportText();
    if(!txt){showToast('Aucun plan à exporter.');return;}
    if(navigator.clipboard){
      if(navigator.clipboard.writeText){
        navigator.clipboard.writeText(txt).then(function(){showToast('✔ Texte copié dans le presse-papiers');}).catch(function(){fallbackCopy(txt);});
      } else {fallbackCopy(txt);}
    } else {fallbackCopy(txt);}
  }

  function exportJSON(){
    var snap = cwSnapshotPreset();
    if(!snap){ showToast('Aucun plan à exporter.'); return; }
    var blob = new Blob([JSON.stringify(snap, null, 2)], {type:'application/json'});
    var url = URL.createObjectURL(blob);
    var a = document.createElement('a');
    a.href = url;
    a.download = (snap.nom || 'plan-clavier').replace(/[^a-zA-Z0-9_\u00C0-\u017E\-]/g, '-') + '.json';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    showToast('\u2714 JSON téléchargé');
  }

  function fallbackCopy(text){
    var ta = document.createElement('textarea');
    ta.value = text;
    ta.style.position = 'fixed';
    ta.style.left = '-9999px';
    document.body.appendChild(ta);
    ta.select();
    try { document.execCommand('copy'); showToast('JSON copié \u2714'); }
    catch(e){ showToast('Impossible de copier. Copiez manuellement.'); }
    document.body.removeChild(ta);
  }

  /* ═══════════════════════════════════════════════════
     TEMPLATE MANAGEMENT (localStorage)
     ═══════════════════════════════════════════════════ */
  function getSavedPlans(){
    try { return JSON.parse(localStorage.getItem(LS_KEY)) || {}; }
    catch(e){ return {}; }
  }
  function setSavedPlans(plans){
    localStorage.setItem(LS_KEY, JSON.stringify(plans));
  }
  function refreshSavedList(){
    var sel = $('cw-saved-sel');
    sel.innerHTML = '<option value="">— Aucun plan sauvegardé —</option>';
    var plans = getSavedPlans();
    var keys = Object.keys(plans).sort();
    if(keys.length){
      sel.innerHTML = '<option value="">— Choisir un plan —</option>';
      keys.forEach(function(k){
        var opt = document.createElement('option');
        opt.value = k;
        opt.textContent = k;
        sel.appendChild(opt);
      });
    }
  }
  function savePlan(){
    var snap = cwSnapshotPreset();
    if(!snap){ showToast('Aucun plan à sauvegarder.'); return; }
    var suggested = prompt('Nom du plan :', snap.nom);
    if(!suggested) return;
    snap.nom = suggested;
    currentPreset.id = snap.id;
    var plans = getSavedPlans();
    plans[suggested] = snap;
    setSavedPlans(plans);
    refreshSavedList();
    showToast('Plan « '+suggested+' » sauvegardé \u2714');
  }
  function loadPlan(){
    var key = $('cw-saved-sel').value;
    if(!key){ showToast('Sélectionnez un plan.'); return; }
    var plans = getSavedPlans();
    if(!plans[key]){ showToast('Plan introuvable.'); return; }
    currentPreset = JSON.parse(JSON.stringify(plans[key]));
    // Update UI fields
    $('cw-nom-instrument').value = currentPreset.nom || currentPreset.label || '';
    $('cw-nom-joueur').value = currentPreset.joueur || '';
    $('cw-preset-sel').value = '';
    renderAll();
    showToast('Plan « '+key+' » chargé \u2714');
  }
  function deletePlan(){
    var key = $('cw-saved-sel').value;
    if(!key){ showToast('Sélectionnez un plan.'); return; }
    if(!confirm('Supprimer « '+key+' » ?')) return;
    var plans = getSavedPlans();
    delete plans[key];
    setSavedPlans(plans);
    refreshSavedList();
    showToast('Plan supprimé.');
  }

  function cwNormalisePreset(json){
    if(Array.isArray(json)) json = json[0];
    return json || null;
  }

  function loadFromFile(event){
    var file = event.target.files[0];
    if(!file) return;
    event.target.value = ''; // reset so same file can be re-loaded
    var reader = new FileReader();
    reader.onload = function(e){
      try {
        var json = JSON.parse(e.target.result);
        json = cwNormalisePreset(json);
        if(!json){ showToast('Format non reconnu'); return; }
        currentPreset = json;
        $('cw-preset-sel').value = '';
        $('cw-nom-instrument').value = json.nom || '';
        $('cw-nom-joueur').value = json.joueur || '';
        renderAll();
        showToast('\u2714 Fichier chargé');
      } catch(ex){ showToast('Erreur: JSON invalide'); }
    };
    reader.readAsText(file);
  }

  /* ═══════════════════════════════════════════════════
     DARK/LIGHT TOGGLE
     ═══════════════════════════════════════════════════ */
  function toggleTheme(){
    darkMode = !darkMode;
    widget.classList.toggle('cw-dark', darkMode);
    $('cw-theme-toggle').textContent = darkMode ? '☀️' : '🌙';
    if(currentPreset) renderAll();
  }

  /* ═══════════════════════════════════════════════════
     COLLAPSIBLE SECTIONS
     ═══════════════════════════════════════════════════ */
  function initCollapsible(){
    widget.querySelectorAll('.cw-section-head[data-toggle]').forEach(function(head){
      head.addEventListener('click', function(){
        var targetId = head.getAttribute('data-toggle');
        var body = $(targetId);
        if(!body) return;
        var hidden = body.classList.contains('cw-hidden');
        body.classList.toggle('cw-hidden', !hidden);
        head.textContent = head.textContent.replace(/^[▶▼]/, hidden ? '▼' : '▶');
      });
    });
  }


  /* ═══════════════════════════════════════════════════
     SVG CLICK-TO-EDIT — handled by openNotePicker()
     ═══════════════════════════════════════════════════ */

  function bindSVGEditing(){
    if(!widget) return;
    widget.querySelectorAll('.cw-edit-btn').forEach(function(g){
      g.addEventListener('click', function(e){
        e.stopPropagation();
        openNotePicker(g.getAttribute('data-edit'), e);
      });
    });
  }

  /* ═══════════════════════════════════════════════════
     INIT
     ═══════════════════════════════════════════════════ */
  function init(){
    mergeCustomPresets();
    populatePresetSelector();
    initCollapsible();
    refreshSavedList();

    // Admin block: visible only when URL hash contains #cw-admin or widget has data-cw-admin attr
    var widget = document.querySelector('.clavier-widget');
    var adminBlock = $('cw-admin-block');
    if(adminBlock){
      var isAdmin = window.location.hash === '#cw-admin';
      if(!isAdmin){ isAdmin = widget ? widget.hasAttribute('data-cw-admin') : false; }
      if(isAdmin){ adminBlock.style.display = 'block'; }
    }

    // Event listeners
    $('cw-preset-sel').addEventListener('change', function(){
      loadPreset(this.value);
    });
    $('cw-nom-instrument').addEventListener('input', function(){
      if(currentPreset) renderAll();
    });
    $('cw-nom-joueur').addEventListener('input', function(){
      if(currentPreset) renderAll();
    });
    $('cw-theme-toggle').addEventListener('click', toggleTheme);
    $('cw-export-btn').addEventListener('click', exportText);
    $('cw-export-json-btn').addEventListener('click', exportJSON);
    $('cw-import-file-btn').addEventListener('click', function(){ $('cw-file-input').click(); });
    $('cw-file-input').addEventListener('change', loadFromFile);
    $('cw-save-btn').addEventListener('click', savePlan);
    $('cw-load-btn').addEventListener('click', loadPlan);
    $('cw-delete-btn').addEventListener('click', deletePlan);
    $('cw-copy-preset-btn').addEventListener('click', exportAsPreset);

    // Edit popup events
    $('cw-np-close').addEventListener('click', closeNotePicker);
    $('cw-np-clear-btn').addEventListener('click', cwPickerClear);
    $('cw-np-apply-btn').addEventListener('click', cwPickerApply);
    $('cw-np-tab-push').addEventListener('click', function(){ if(_cwEdit) cwSwitchSlot('push'); });
    $('cw-np-tab-pull').addEventListener('click', function(){ if(_cwEdit) cwSwitchSlot('pull'); });
    // Click-outside: close picker (inspired by offline tool — check style.display, not class)
    document.addEventListener('click', function(ev){
      var picker = $('cw-note-picker');
      if(picker && picker.style.display === 'block'){
        if(!picker.contains(ev.target)){
          if(!ev.target.closest('.cw-edit-btn') && !ev.target.closest('input[data-section]')){
            closeNotePicker();
          }
        }
      }
    });
    // Note table inputs → open picker on click (inspired by offline tool)
    $('cw-rh-tables').addEventListener('click', function(e){
      var inp = e.target.closest('input[data-section="rh"]');
      if(inp && currentPreset){
        var ri = parseInt(inp.getAttribute('data-row'), 10);
        var bi = parseInt(inp.getAttribute('data-btn'), 10);
        var dir = inp.getAttribute('data-dir');
        e.stopPropagation();
        openNotePicker('rh:'+ri+':'+bi, e, dir === 'p' ? 'push' : 'pull');
      }
    });
    $('cw-lh-tables').addEventListener('click', function(e){
      var inp = e.target.closest('input[data-section]');
      if(inp && currentPreset){
        var sec = inp.getAttribute('data-section');
        var bi  = parseInt(inp.getAttribute('data-btn'), 10);
        var dir = inp.getAttribute('data-dir');
        e.stopPropagation();
        openNotePicker(sec+':'+bi, e, dir === 'p' ? 'push' : 'pull');
      }
    });
    initNotePicker();
  }

  // Run init when DOM is ready
  if(document.readyState === 'loading'){
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

})();
