  /* ── Disposition dérivée du plan sélectionné ── */
  function getDispositionFromPlan() {
    if (!planData || !planCategory) return '\u2013';
    if (planCategory === '3 rangs 12b') {
      /* Présets standard ("33 boutons") = Heim ; Spéciaux = label contient l'info */
      if (planLabel &#038;&#038; planLabel.indexOf('33 boutons') >= 0) return 'Heim (standard)';
      return planLabel || planCategory;
    }
    if (planCategory === '2 rangs 8b') {
      /* Irlandais vs standard 2 rangs */
      return (planLabel && planLabel.toLowerCase().indexOf('irlandais') >= 0) ? 'Irish' : '\u2013';
    }
    /* 3 rangs 18b, 3 rangs 24b — le label contient déjà l'info de disposition */
    return planLabel || planCategory;
  }

  /* Sync nbBoutonsMD and nbBoutonsMG from loaded plan data */
  function syncPlanToForm(inst) {
    if (!inst) return;
    var nRows = inst.nbRangees || (inst.rangees ? inst.rangees.length : 0);
    if (nRows === 2 || nRows === 3) {
      var val = nRows === 2 ? '21' : '33';
      _syncingFromPlan = true;
      nbMD.value = val;
      nbMD.dispatchEvent(new Event('change'));
      _syncingFromPlan = false;
    }
    var mgSel = document.getElementById('nbBoutonsMG');
    if (mgSel) { if (inst.notes) {
      var bl = inst.notes.basses ? inst.notes.basses.length : 0;
      var al = inst.notes.accords ? inst.notes.accords.length : 0;
      var lhNb = bl + al;
      if (lhNb > 0) {
        var best = '8b';
        if (lhNb >= 22) best = '24b';
        else if (lhNb >= 16) best = '18b';
        else if (lhNb >= 10) best = '12b';
        mgSel.value = best;
      }
    } }
  }

  function showPlanPreview(inst, label) {
    var preview = document.getElementById('planPreview');
    var info = document.getElementById('planPreviewInfo');
    var svgContainer = document.getElementById('planSVG');
    var nbRows = inst.nbRangees || (inst.rangees ? inst.rangees.length : '?');
    var totalBtns = 0;
    if (inst.rangees) inst.rangees.forEach(function (r) { totalBtns += (r.nb || 0); });
    var lhBtns = 0;
    if (inst.lhRows) { if (inst.lhRows.length) {
      inst.lhRows.forEach(function (r) { lhBtns += (r.nb || 0) * 2; });
    } }
    if (!lhBtns) { if (inst.notes) {
      var bl2 = inst.notes.basses ? inst.notes.basses.length : 0;
      var al2 = inst.notes.accords ? inst.notes.accords.length : 0;
      lhBtns = bl2 + al2;
    } }
    info.innerHTML = '<strong>' + (label || inst.nom || '(sans nom)') + '</strong><br>'
      + 'MD : ' + nbRows + ' rang\u00e9es \u00b7 ' + totalBtns + ' boutons'
      + (lhBtns ? '<br>MG : ' + lhBtns + ' boutons' : '');
    preview.classList.add('visible');
    try {
      planSvgData = buildSVG_PlanLite.presetToData(inst);
      var svgStr = buildSVG_PlanLite.render(planSvgData);
      svgContainer.innerHTML = svgStr;
      var svgEl = svgContainer.querySelector('svg');
      if (svgEl) { svgEl.setAttribute('width', '100%'); svgEl.setAttribute('height', 'auto'); svgEl.style.maxWidth = '100%'; }
      /* Mise à jour de l'aperçu inline dans la section MD */
      var inlineContainer = document.getElementById('planSVGInline');
      var planPreviewRH = document.getElementById('planPreviewRH');
      if (inlineContainer) {
        inlineContainer.innerHTML = svgStr;
        var svgEl2 = inlineContainer.querySelector('svg');
        if (svgEl2) { svgEl2.setAttribute('width', '100%'); svgEl2.setAttribute('height', 'auto'); svgEl2.style.maxWidth = '100%'; }

    } catch (svgErr) {
      svgContainer.innerHTML = '<p style="color:#999;font-size:.85rem;">Aper\u00e7u visuel non disponible pour ce format.</p>';
    }
    /* Auto-synchro nbBoutonsMD/MG depuis le plan */
    syncPlanToForm(inst);
  }

  function onPresetSelect(id) {
    var loadState = document.getElementById('planPresetLoadState');
    planData = null; planSvgData = null; planLabel = '';
    var preview = document.getElementById('planPreview');
    var svgContainer = document.getElementById('planSVG');
    if (!id) {
      preview.classList.remove('visible');
      svgContainer.innerHTML = '';
      updateSummary();
      return;
    }
    if (loadState) loadState.textContent = 'Chargement\u2026';
    fetch(PRESETS_BASE_URL + id + '.json')
      .then(function (r) { if (!r.ok) throw new Error('HTTP ' + r.status); return r.json(); })
      .then(function (inst) {
        planData = inst;
        var sel = document.getElementById('planPresetSelect');
        planLabel = sel ? sel.options[sel.selectedIndex].text : (inst.nom || id);
        /* Look up the category from the presets index */
        var meta = allPresets.find(function (p) { return p.id === id; });
        planCategory = meta ? meta.category : '';
        if (loadState) loadState.textContent = '';
        showPlanPreview(inst, planLabel);
        updateSummary();
      })
      .catch(function () {
        if (loadState) loadState.textContent = '\u26a0 Impossible de charger le pr\u00e9set.';
        planData = null;
        if (preview) preview.classList.remove('visible');
        if (svgContainer) svgContainer.innerHTML = '';
        updateSummary();
      });
  }

  /* ── Accordion Visual Preview ── */
  /* ── Aperçu accordéon — calques Nano Banana ─────────────────────────────
   * Chaque calque charge son PNG selon l'option choisie.
   * Convention : ACC_IMG_BASE + data-img-base + '-' + slug + '.png'
   * ex : img/apercu/caisse-md-noyer.png
   *
   * Fallback : background-color visible si le PNG n'existe pas encore.
   */

  var ACC_IMG_BASE = 'img/apercu/';
  if (typeof STAGE_CONFIG !== 'undefined') { if (STAGE_CONFIG.apercuBaseUrl) { ACC_IMG_BASE = STAGE_CONFIG.apercuBaseUrl; } }

  /* Correspondance valeur option → slug fichier image.
   * Note : '\u00c9rable' (Érable, avec accent) = bois du clavier/grille.
   *        'Erable' (sans accent) = valeur du champ boutonsMD. */
  var VALUE_TO_SLUG = {
    'Noyer'          : 'noyer',
    '\u00c9rable'    : 'erable',
    'Cerisier'       : 'cerisier',
    'Nacrine noire'  : 'nacrine-noire',
    'Nacrine blanche': 'nacrine-blanche',
    'Erable'         : 'erable',
    'Bleu'           : 'bleu',
    'Rouge'          : 'rouge',
    'Orange'         : 'orange',
    'Noir'           : 'noir',
    'Cuir Nat.'      : 'cuir-nat'
  };

  /* Couleurs de fallback (affichées avant que les PNG soient disponibles) */
  var WOOD_COLORS    = { 'Noyer': '#5C3D2E', '\u00c9rable': '#C8A96E', 'Cerisier': '#8B4513' };
  var BUTTON_COLORS  = { 'Nacrine noire': '#1a1a1a', 'Nacrine blanche': '#F5F0E8', 'Noyer': '#6B4226', 'Erable': '#C8A96E' };
  var BELLOWS_COLORS = { 'Bleu': '#2e6cb5', 'Rouge': '#c0392b', 'Orange': '#e67e22', 'Noir': '#222' };
  var STRAP_COLORS   = { 'Bleu': '#2e6cb5', 'Rouge': '#c0392b', 'Cuir Nat.': '#B5885A', 'Noir': '#222' };

  /* setLayerImage: met à jour un calque selon la valeur de l'option.
   * - Charge le PNG correspondant (probe async — si absent, background-color visible)
   * - Classe has-img ajoutée quand l'image est disponible (désactive clip-path) */
  function setLayerImage(el, colorMap, value) {
    if (el) {
      if (value) {
        /* Toujours appliquer la couleur de fallback */
        if (colorMap) {
          if (colorMap[value]) {
            el.style.backgroundColor = colorMap[value];
          }
        }
        /* Construire l'URL et tenter le chargement */
        var base = el.getAttribute('data-img-base');
        var slug = VALUE_TO_SLUG[value];
        if (base) {
          if (slug) {
            var src = ACC_IMG_BASE + base + '-' + slug + '.png';
            var probe = new Image();
            probe.onload = function() {
              el.style.backgroundImage = 'url(' + src + ')';
              el.classList.add('has-img');
              el.classList.remove('no-img');
            };
            probe.onerror = function() {
              el.style.backgroundImage = '';
              el.classList.remove('has-img');
              el.classList.add('no-img');
            };
            probe.src = src;
          }
        }
      }
    }
  }

  function updateAccordionPreview() {

    var boisGrille  = document.getElementById('boisGrille').value || boisClavier;
    var boutons     = document.getElementById('boutonsMD').value;
    var soufflet    = getRadioVal('couleurSoufflet');
    var sangles     = document.getElementById('couleurSangles').value;

    setLayerImage(document.getElementById('accGrille'),    WOOD_COLORS,    boisGrille);

  /* ── PDF Generation (returns jsPDF doc) ── */
  function buildPDF() {
    var jsPDF = window.jspdf.jsPDF;
    var doc = new jsPDF({ unit: 'mm', format: 'a4' });
    var W = 210;
    var H = 297;
    var MARGIN = 18;
    var CW = W - 2 * MARGIN;
    var PAGE_LIMIT = H - MARGIN - 14; /* max y before forcing page break */
    var y = MARGIN;

    /* Colors */
    var woodDark = [62, 39, 35];
    var gold = [212, 160, 23];
    var grey = [120, 120, 120];

    /* Header bar */
    doc.setFillColor.apply(doc, woodDark);
    doc.rect(0, 0, W, 36, 'F');
    doc.setTextColor(255, 255, 255);
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(18);
    doc.text('Formulaire d\'inscription', W / 2, 16, { align: 'center' });
    doc.setFontSize(11);
    doc.setFont('helvetica', 'normal');
    doc.text('Stage de lutherie \u2013 Accord\u00e9on diatonique ' + ANNEE, W / 2, 26, { align: 'center' });
    y = 44;

    /* Helper: section title */
    function sectionTitle(title) {
      if (y > PAGE_LIMIT) { doc.addPage(); y = MARGIN; }
      doc.setFillColor.apply(doc, gold);
      doc.rect(MARGIN, y, CW, 8, 'F');
      doc.setTextColor.apply(doc, woodDark);
      doc.setFont('helvetica', 'bold');
      doc.setFontSize(11);
      doc.text(title, MARGIN + 4, y + 5.5);
      y += 12;
    }

    /* Helper: line */
    function addLine(label, value) {
      if (y > PAGE_LIMIT) { doc.addPage(); y = MARGIN; }
      doc.setFont('helvetica', 'normal');
      doc.setFontSize(9.5);
      doc.setTextColor.apply(doc, grey);
      doc.text(label, MARGIN, y);
      doc.setTextColor.apply(doc, woodDark);
      doc.setFont('helvetica', 'bold');
      doc.text(String(value || '\u2013'), MARGIN + 60, y);
      y += 6.5;
    }

    /* 1. Coordonnées */
    sectionTitle('Coordonn\u00e9es personnelles');
    addLine('Nom pr\u00e9nom', document.getElementById('nom').value);
    addLine('Date de naissance', document.getElementById('dateNaissance').value);
    addLine('Adresse', document.getElementById('adresse').value);
    addLine('CP + Ville', document.getElementById('cp').value + ' ' + document.getElementById('ville').value);
    addLine('T\u00e9l\u00e9phone', document.getElementById('telephone').value);
    addLine('Email', document.getElementById('email').value);
    y += 4;

    /* 2. Session */
    sectionTitle('Session choisie');
    var sessVal = getRadioVal('session');
    addLine('Session', SESSION_LABELS[sessVal] || '\u2013');
    y += 4;

    /* 3. Modèle */
    sectionTitle('Mod\u00e8le');
    addLine('Mod\u00e8le', getRadioVal('modele'));
    var pdfAncheChecked = document.getElementById('ancheAMano').checked;
    addLine('Anche a mano Blue Star', pdfAncheChecked ? 'Oui (+15 %)' : 'Non');
    y += 4;

    /* 4. Main Droite */
    sectionTitle('Caisse Main Droite');
    addLine('Nb boutons', selVal('nbBoutonsMD') || '\u2013');
    addLine('Disposition', getDispositionFromPlan());
    addLine('Tonalit\u00e9', selVal('tonalite') || '\u2013');
    addLine('Voix MD', selVal('voix1') || '\u2013');
    var pdfUp2V = document.getElementById('upgrade2V').checked;
    addLine('Upgrade 2V registres', pdfUp2V ? 'Oui (+8 % \u2014 2\u1d49 stage n\u00e9cessaire)' : 'Non');
    if (pdfUp2V) addLine('Voix 2 MD', selVal('voix2') || '\u2013');
    addLine('Accordage', selVal('accordage') || '\u2013');
    addLine('Boutons', selVal('boutonsMD') || '\u2013');
    addLine('Bois clavier', selVal('boisClavier') || '\u2013');
    addLine('Bois grille/caisse', selVal('boisGrille') || '\u2013');
    var pdfGrille = document.getElementById('numGrille').value;
    addLine('Grille', pdfGrille ? 'Grille n\u00b0' + pdfGrille : '\u2013');
    addLine('Marquage boutons', document.getElementById('marquageMD').value || '\u2013');
    var remMD = document.getElementById('remarquesMD').value;
    if (remMD) addLine('Remarques MD', remMD);
    y += 4;

    /* 5. Main Gauche */
    sectionTitle('Caisse Main Gauche');
    addLine('Nb boutons basses', selVal('nbBoutonsMG') || '\u2013');
    addLine('Couleur sangles', selVal('couleurSangles') || '\u2013');
    var remMG = document.getElementById('remarquesMG').value;
    if (remMG) addLine('Remarques MG', remMG);
    y += 4;

    /* 6. Soufflet */
    sectionTitle('Le Soufflet');
    addLine('Couleur soufflet', getRadioVal('couleurSoufflet'));
    y += 4;

    /* 7. Plan de clavier */
    if (planData) {
      sectionTitle('Plan de clavier');
      addLine('Plan', planLabel || planData.nom || '(personnalis\u00e9)');
      if (planData.rangees) {
        planData.rangees.forEach(function (r, i) {
          addLine('Rang\u00e9e ' + (i + 1), (r.nom || '') + ' \u2013 ' + (r.nb || 0) + ' boutons');
        });
      }
      y += 4;
    }

    y += 2;

    /* Total */
    var modEl = form.querySelector('input[name="modele"]:checked');
    if (modEl) {
      var basePrice = parseInt(modEl.dataset.price, 10);
      var price = computePrice(basePrice);
      var deposit = Math.round(price * DEPOSIT_RATE);
      if (y > PAGE_LIMIT) { doc.addPage(); y = MARGIN; }
      doc.setFillColor(255, 248, 225);
      var boxH = 28;
      var surchargeLines = [];
      if (pdfAncheChecked) surchargeLines.push('Anche a mano Blue Star +' + Math.round(basePrice * (ANCHE_SURCHARGE - 1)).toLocaleString('fr-FR') + ' \u20ac');
      if (pdfUp2V) surchargeLines.push('2 Voix +' + Math.round(basePrice * (UPGRADE_2V_SURCHARGE - 1)).toLocaleString('fr-FR') + ' \u20ac');
      if (surchargeLines.length) boxH += 8 * surchargeLines.length;
      doc.roundedRect(MARGIN, y, CW, boxH, 3, 3, 'F');
      doc.setTextColor.apply(doc, woodDark);
      doc.setFont('helvetica', 'bold');
      doc.setFontSize(14);
      doc.text('Prix total : ' + price.toLocaleString('fr-FR') + ' \u20ac', W / 2, y + 9, { align: 'center' });
      var ySub = y + 17;
      if (surchargeLines.length) {
        doc.setFontSize(9);
        doc.setTextColor(150, 120, 50);
        doc.text('(dont ' + surchargeLines.join(', ') + ')', W / 2, ySub, { align: 'center' });
        ySub += 8 * surchargeLines.length;
      }
      doc.setFontSize(10);
      doc.setTextColor.apply(doc, gold);
      doc.text('Acompte \u00e0 l\'inscription : ' + deposit.toLocaleString('fr-FR') + ' \u20ac', W / 2, ySub, { align: 'center' });
      y += boxH + 4;
    }

    /* Terms */
    if (y > PAGE_LIMIT) { doc.addPage(); y = MARGIN; }
    doc.setFontSize(8);
    doc.setFont('helvetica', 'normal');
    doc.setTextColor(150, 150, 150);
    var terms = 'Ce bon de commande vous engage. Pour \u00eatre valide, ce bon doit \u00eatre accompagn\u00e9 d\'un acompte de 40 %. '
      + 'Il ne pourra \u00eatre restitu\u00e9 si le client annule son inscription apr\u00e8s le d\u00e9but de la fabrication (2 mois avant le stage).';
    var splitTerms = doc.splitTextToSize(terms, CW);
    doc.text(splitTerms, MARGIN, y);
    y += splitTerms.length * 4 + 6;

    /* Footer */
    doc.setDrawColor.apply(doc, gold);
    doc.setLineWidth(0.5);
    doc.line(MARGIN, y, W - MARGIN, y);
    y += 5;
    doc.setFontSize(8);
    doc.setTextColor.apply(doc, grey);
    doc.text('Ewen Daviau \u2013 9 rue Fernand de Magellan, 44600 Saint-Nazaire', W / 2, y, { align: 'center' });
    y += 4;
    doc.text('contact@ewendaviau.com \u00b7 ewendaviau.com', W / 2, y, { align: 'center' });

    return doc;
  }
